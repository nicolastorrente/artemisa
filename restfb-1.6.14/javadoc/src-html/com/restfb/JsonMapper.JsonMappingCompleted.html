023package com.restfb;
024
025import static java.lang.annotation.ElementType.METHOD;
026import static java.lang.annotation.RetentionPolicy.RUNTIME;
027
028import java.lang.annotation.Retention;
029import java.lang.annotation.Target;
030import java.util.List;
031
032import com.restfb.exception.FacebookJsonMappingException;
033
034/**
035 * Specifies how a Facebook JSON-to-Java (and vice-versa) mapper must operate.
036 * <p>
037 * Note that implementors must be able to handle illegal JSON in {@link #toJavaObject(String, Class)} and
038 * {@link #toJavaList(String, Class)} in order to correctly process Facebook responses. For example, the
039 * {@code Users.getLoggedInUser} Legacy API call returns a value like {@code 1240077}, which is not valid JSON.
040 * 
041 * @author <a href="http://restfb.com">Mark Allen</a>
042 */
043public interface JsonMapper {
044  /**
045   * Given a JSON string, create and return a new instance of a corresponding Java object of type {@code type}.
046   * <p>
047   * The Java {@code type} must have a no-argument constructor.
048   * 
049   * @param <T>
050   *          Java type to map to.
051   * @param json
052   *          The JSON to be mapped to a Java type.
053   * @param type
054   *          Java type token.
055   * @return A Java object (of type {@code type}) representation of the JSON input.
056   * @throws FacebookJsonMappingException
057   *           If an error occurs while mapping JSON to Java.
058   */
059  <T> T toJavaObject(String json, Class<T> type);
060
061  /**
062   * Given a JSON string, create and return a new instance of a corresponding Java {@link java.util.List} which contains
063   * elements of type {@code type}.
064   * <p>
065   * The Java {@code type} must have a no-argument constructor.
066   * 
067   * @param <T>
068   *          Java type to map to for each element of the list.
069   * @param json
070   *          The JSON to be mapped to a Java type.
071   * @param type
072   *          Java type token.
073   * @return A Java object (of type {@code List} which contains elements of type {@code type}) representation of the
074   *         JSON input.
075   * @throws FacebookJsonMappingException
076   *           If an error occurs while mapping JSON to Java.
077   */
078  <T> List<T> toJavaList(String json, Class<T> type);
079
080  /**
081   * Given a Java {@code object}, create and return a JSON string that represents it.
082   * <p>
083   * The {@code object}'s properties will be traversed recursively, allowing for arbitrarily complex JSON generation.
084   * 
085   * @param object
086   *          The Java object to map to JSON. Can be a Javabean, {@link java.util.List}, or {@link java.util.Map}.
087   * @return A JSON string.
088   * @throws FacebookJsonMappingException
089   *           If an error occurs while mapping Java to JSON.
090   * @since 1.4
091   */
092  String toJson(Object object);
093
094  /**
095   * Given a Java {@code object}, create and return a JSON string that represents it.
096   * <p>
097   * The {@code object}'s properties will be traversed recursively, allowing for arbitrarily complex JSON generation.
098   * 
099   * @param object
100   *          The Java object to map to JSON. Can be a Javabean, {@link java.util.List}, or {@link java.util.Map}.
101   * @param ignoreNullValuedProperties
102   *          If {@code true}, no Javabean properties with {@code null} values will be included in the generated JSON.
103   * @return A JSON string.
104   * @throws FacebookJsonMappingException
105   *           If an error occurs while mapping Java to JSON.
106   * @since 1.6.5
107   */
108  String toJson(Object object, boolean ignoreNullValuedProperties);
109
110  /**
111   * If you apply this annotation to a method of a type mapped by {@code JsonMapper}, it will be called after the
112   * mapping operation has been completed.
113   * <p>
114   * The method you specify must take 0 parameters or a single {@code JsonMapper} parameter. Any other signature will
115   * cause a {@code FacebookJsonMappingException} to be thrown.
116   * <p>
117   * This is useful if you'd like to perform a custom post-mapping task, like massaging the data Facebook returns or
118   * custom mapping of fields {@code JsonMapper} isn't equipped to handle yet.
119   * 
120   * @author <a href="http://restfb.com">Mark Allen</a>
121   * @since 1.6.11
122   */
123  @Target({ METHOD })
124  @Retention(RUNTIME)
125  public @interface JsonMappingCompleted {}
126}
