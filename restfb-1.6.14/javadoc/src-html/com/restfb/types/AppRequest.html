023package com.restfb.types;
024
025import static com.restfb.util.DateUtils.toDateFromLongFormat;
026
027import java.util.Date;
028
029import com.restfb.Facebook;
030
031/**
032 * Represents the <a href="https://developers.facebook.com/docs/reference/api/user/#apprequests" >App Request Graph API
033 * type</a>.
034 * 
035 * @author <a href="http://restfb.com">Mark Allen</a>
036 * @since 1.6.10
037 */
038public class AppRequest extends FacebookType {
039  @Facebook
040  private Application application;
041
042  @Facebook
043  private NamedFacebookType to;
044
045  @Facebook
046  private NamedFacebookType from;
047
048  @Facebook
049  private String message;
050
051  @Facebook("created_time")
052  private String createdTime;
053
054  private static final long serialVersionUID = 1L;
055
056  /**
057   * Represents the <a href="http://developers.facebook.com/docs/reference/api/page">Cover Graph API type</a>.
058   * 
059   * @author <a href="http://restfb.com">Mark Allen</a>
060   * @since 1.6.10
061   */
062  public static class Application extends NamedFacebookType {
063    @Facebook("canvas_name")
064    private String canvasName;
065
066    @Facebook
067    private String namespace;
068
069    private static final long serialVersionUID = 1L;
070
071    /**
072     * The application's canvas name.
073     * 
074     * @return The application's canvas name.
075     */
076    public String getCanvasName() {
077      return canvasName;
078    }
079
080    /**
081     * The application's namespace.
082     * 
083     * @return The application's namespace.
084     */
085    public String getNamespace() {
086      return namespace;
087    }
088  }
089
090  /**
091   * App associated with the request.
092   * 
093   * @return App associated with the request.
094   */
095  public Application getApplication() {
096    return application;
097  }
098
099  /**
100   * The recipient user associated with the request.
101   * 
102   * @return The recipient user associated with the request.
103   */
104  public NamedFacebookType getTo() {
105    return to;
106  }
107
108  /**
109   * The sender user associated with the request.
110   * 
111   * @return The sender user associated with the request.
112   */
113  public NamedFacebookType getFrom() {
114    return from;
115  }
116
117  /**
118   * A string describing the request.
119   * 
120   * @return A string describing the request.
121   */
122  public String getMessage() {
123    return message;
124  }
125
126  /**
127   * Timestamp when the request was created.
128   * 
129   * @return Timestamp when the request was created.
130   */
131  public Date getCreatedTime() {
132    return toDateFromLongFormat(createdTime);
133  }
134}
