023package com.restfb.types;
024
025import static com.restfb.util.DateUtils.toDateFromLongFormat;
026import static java.util.Collections.unmodifiableList;
027
028import java.util.ArrayList;
029import java.util.Date;
030import java.util.List;
031
032import com.restfb.Facebook;
033
034/**
035 * Represents the <a href="http://developers.facebook.com/docs/reference/api/note">Note Graph API type</a>.
036 * 
037 * @author <a href="http://restfb.com">Mark Allen</a>
038 * @since 1.5
039 */
040public class Note extends FacebookType {
041  @Facebook
042  private NamedFacebookType from;
043
044  @Facebook
045  private String subject;
046
047  @Facebook
048  private String message;
049
050  @Facebook
051  private String icon;
052
053  @Facebook("created_time")
054  private String createdTime;
055
056  @Facebook("updated_time")
057  private String updatedTime;
058
059  @Facebook("comments")
060  // It's possible for Facebook to return {"count":0} instead of a list of comments for whatever reason.
061  // So we have this throwaway field to hold that result. There is no need to expose this through the public API; it's
062  // here to let the JSON mapper work without erroring out.
063  private String commentsAsObject;
064
065  @Facebook("comments")
066  private List<Comment> comments = new ArrayList<Comment>();
067
068  private static final long serialVersionUID = 1L;
069
070  /**
071   * The ID of the user who posted the note.
072   * 
073   * @return The ID of the user who posted the note.
074   */
075  public NamedFacebookType getFrom() {
076    return from;
077  }
078
079  /**
080   * The title of the note.
081   * 
082   * @return The title of the note.
083   */
084  public String getSubject() {
085    return subject;
086  }
087
088  /**
089   * The note content, an HTML string.
090   * 
091   * @return The note content, an HTML string.
092   */
093  public String getMessage() {
094    return message;
095  }
096
097  /**
098   * The note icon.
099   * 
100   * @return The note icon.
101   */
102  public String getIcon() {
103    return icon;
104  }
105
106  /**
107   * The time the note was initially published.
108   * 
109   * @return The time the note was initially published.
110   */
111  public Date getCreatedTime() {
112    return toDateFromLongFormat(createdTime);
113  }
114
115  /**
116   * The time the note was last updated.
117   * 
118   * @return The time the note was last updated.
119   */
120  public Date getUpdatedTime() {
121    return toDateFromLongFormat(updatedTime);
122  }
123
124  /**
125   * Comments made on the note.
126   * 
127   * @return Comments made on the note.
128   * @since 1.6.10
129   */
130  public List<Comment> getComments() {
131    return unmodifiableList(comments);
132  }
133}
