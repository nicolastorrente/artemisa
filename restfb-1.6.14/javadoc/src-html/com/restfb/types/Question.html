023package com.restfb.types;
024
025import static com.restfb.util.DateUtils.toDateFromLongFormat;
026import static java.util.Collections.unmodifiableList;
027
028import java.util.ArrayList;
029import java.util.Date;
030import java.util.List;
031
032import com.restfb.Facebook;
033
034/**
035 * Represents the <a href="http://developers.facebook.com/docs/reference/api/question">Question Graph API type</a>.
036 * 
037 * @author <a href="http://restfb.com">Mark Allen</a>
038 * @since 1.6.10
039 */
040public class Question extends FacebookType {
041  @Facebook
042  private NamedFacebookType from;
043
044  @Facebook
045  private String question;
046
047  @Facebook("created_time")
048  private String createdTime;
049
050  @Facebook("updated_time")
051  private String updatedTime;
052
053  @Facebook
054  private List<QuestionOption> options = new ArrayList<QuestionOption>();
055
056  private static final long serialVersionUID = 1L;
057
058  /**
059   * User who asked the question.
060   * 
061   * @return User who asked the question.
062   */
063  public NamedFacebookType getFrom() {
064    return from;
065  }
066
067  /**
068   * Text of the question.
069   * 
070   * @return Text of the question.
071   */
072  public String getQuestion() {
073    return question;
074  }
075
076  /**
077   * Time when question was created.
078   * 
079   * @return Time when question was created.
080   */
081  public Date getCreatedTime() {
082    return toDateFromLongFormat(createdTime);
083  }
084
085  /**
086   * Time when question was last updated.
087   * 
088   * @return Time when question was last updated.
089   */
090  public Date getUpdatedTime() {
091    return toDateFromLongFormat(updatedTime);
092  }
093
094  /**
095   * The list of options available as answers to the question.
096   * 
097   * @return The list of options available as answers to the question.
098   */
099  public List<QuestionOption> getOptions() {
100    return unmodifiableList(options);
101  }
102}
