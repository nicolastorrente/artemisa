023package com.restfb.types;
024
025import com.restfb.Facebook;
026
027/**
028 * Represents a Facebook URL.
029 * 
030 * @author <a href="http://restfb.com">Mark Allen</a>
031 * @deprecated As of 1.6.10, this type is deprecated and will be removed in a future release.
032 */
033@Deprecated
034public class Url extends NamedFacebookType {
035  @Facebook
036  private Long shares;
037
038  @Facebook
039  private String picture;
040
041  @Facebook
042  private String link;
043
044  @Facebook
045  private String category;
046
047  @Facebook("fan_count")
048  private Long fanCount;
049
050  private static final long serialVersionUID = 1L;
051
052  public Long getShares() {
053    return shares;
054  }
055
056  public String getPicture() {
057    return picture;
058  }
059
060  public String getLink() {
061    return link;
062  }
063
064  public String getCategory() {
065    return category;
066  }
067
068  public Long getFanCount() {
069    return fanCount;
070  }
071}
