023package com.restfb.types;
024
025import static com.restfb.util.DateUtils.toDateFromLongFormat;
026import static java.util.Collections.unmodifiableList;
027
028import java.io.Serializable;
029import java.util.ArrayList;
030import java.util.Date;
031import java.util.List;
032
033import com.restfb.Facebook;
034import com.restfb.util.ReflectionUtils;
035
036/**
037 * Represents the <a href="http://developers.facebook.com/docs/reference/api/photo">Photo Graph API type</a>.
038 * 
039 * @author <a href="http://restfb.com">Mark Allen</a>
040 * @since 1.5
041 */
042public class Photo extends NamedFacebookType {
043  @Facebook
044  private CategorizedFacebookType from;
045
046  @Facebook
047  private String picture;
048
049  @Facebook
050  private String source;
051
052  @Facebook
053  private Integer height;
054
055  @Facebook
056  private Integer width;
057
058  @Facebook
059  private String link;
060
061  @Facebook
062  private String icon;
063
064  @Facebook
065  private Integer position;
066
067  @Facebook("created_time")
068  private String createdTime;
069
070  @Facebook("updated_time")
071  private String updatedTime;
072
073  @Facebook
074  private List<Tag> tags = new ArrayList<Tag>();
075
076  @Facebook
077  private List<Comment> comments = new ArrayList<Comment>();
078
079  @Facebook
080  private List<NamedFacebookType> likes = new ArrayList<NamedFacebookType>();
081
082  @Facebook
083  private List<Image> images = new ArrayList<Image>();
084
085  @Facebook
086  private Place place;
087
088  private static final long serialVersionUID = 1L;
089
090  /**
091   * Represents the <a href="http://developers.facebook.com/docs/reference/api/photo">Tag Graph API type</a>.
092   * 
093   * @author <a href="http://restfb.com">Mark Allen</a>
094   * @since 1.5
095   */
096  public static class Tag extends NamedFacebookType {
097    @Facebook
098    private Integer x;
099
100    @Facebook
101    private Integer y;
102
103    @Facebook("created_time")
104    private String createdTime;
105
106    private static final long serialVersionUID = 1L;
107
108    /**
109     * X coordinate (as a percentage of distance from left vs. width).
110     * 
111     * @return X coordinate (as a percentage of distance from left vs. width).
112     */
113    public Integer getX() {
114      return x;
115    }
116
117    /**
118     * Y coordinate (as a percentage of distance from top vs. height).
119     * 
120     * @return Y coordinate (as a percentage of distance from top vs. height).
121     */
122    public Integer getY() {
123      return y;
124    }
125
126    /**
127     * Date this tag was created.
128     * 
129     * @return Date this tag was created.
130     */
131    public Date getCreatedTime() {
132      return toDateFromLongFormat(createdTime);
133    }
134  }
135
136  /**
137   * Represents the <a href="http://developers.facebook.com/docs/reference/api/photo">Image Graph API type</a>.
138   * 
139   * @author <a href="http://restfb.com">Mark Allen</a>
140   * @since 1.6.5
141   */
142  public static class Image implements Serializable {
143    @Facebook
144    private Integer height;
145
146    @Facebook
147    private Integer width;
148
149    @Facebook
150    private String source;
151
152    private static final long serialVersionUID = 1L;
153
154    /**
155     * @see java.lang.Object#hashCode()
156     */
157    @Override
158    public int hashCode() {
159      return ReflectionUtils.hashCode(this);
160    }
161
162    /**
163     * @see java.lang.Object#equals(java.lang.Object)
164     */
165    @Override
166    public boolean equals(Object that) {
167      return ReflectionUtils.equals(this, that);
168    }
169
170    /**
171     * @see java.lang.Object#toString()
172     */
173    @Override
174    public String toString() {
175      return ReflectionUtils.toString(this);
176    }
177
178    /**
179     * The height of the image in pixels.
180     * 
181     * @return The height of the image in pixels.
182     */
183    public Integer getHeight() {
184      return height;
185    }
186
187    /**
188     * The width of the image in pixels.
189     * 
190     * @return The width of the image in pixels.
191     */
192    public Integer getWidth() {
193      return width;
194    }
195
196    /**
197     * The source URL of the image.
198     * 
199     * @return The source URL of the image.
200     */
201    public String getSource() {
202      return source;
203    }
204  }
205
206  /**
207   * An object containing the name and ID of the user who posted the photo.
208   * 
209   * @return An object containing the name and ID of the user who posted the photo.
210   */
211  public CategorizedFacebookType getFrom() {
212    return from;
213  }
214
215  /**
216   * The album-sized view of the photo.
217   * 
218   * @return The album-sized view of the photo.
219   */
220  public String getPicture() {
221    return picture;
222  }
223
224  /**
225   * The full-sized source of the photo.
226   * 
227   * @return The full-sized source of the photo.
228   */
229  public String getSource() {
230    return source;
231  }
232
233  /**
234   * The height of the photo, in pixels.
235   * 
236   * @return The height of the photo, in pixels.
237   */
238  public Integer getHeight() {
239    return height;
240  }
241
242  /**
243   * The width of the photo, in pixels.
244   * 
245   * @return The width of the photo, in pixels.
246   */
247  public Integer getWidth() {
248    return width;
249  }
250
251  /**
252   * A link to the photo on Facebook.
253   * 
254   * @return A link to the photo on Facebook.
255   */
256  public String getLink() {
257    return link;
258  }
259
260  /**
261   * The icon-sized source of the photo.
262   * 
263   * @return The icon-sized source of the photo.
264   */
265  public String getIcon() {
266    return icon;
267  }
268
269  /**
270   * The position of this photo in the album.
271   * 
272   * @return The position of this photo in the album.
273   * @since 1.6.5
274   * @deprecated Facebook will start returning 0 for this field starting on October 3, 2012.
275   */
276  @Deprecated
277  public Integer getPosition() {
278    return position;
279  }
280
281  /**
282   * The location associated with this photo, if any.
283   * 
284   * @return The place this photo was taken.
285   * @since 1.6.10
286   */
287  public Place getPlace() {
288    return place;
289  }
290
291  /**
292   * The time the photo was initially published.
293   * 
294   * @return The time the photo was initially published.
295   */
296  public Date getCreatedTime() {
297    return toDateFromLongFormat(createdTime);
298  }
299
300  /**
301   * The last time the photo or its caption was updated.
302   * 
303   * @return The last time the photo or its caption was updated.
304   */
305  public Date getUpdatedTime() {
306    return toDateFromLongFormat(updatedTime);
307  }
308
309  /**
310   * An array containing the users and their positions in this photo. The x and y coordinates are percentages from the
311   * left and top edges of the photo, respectively.
312   * 
313   * @return An array containing the users and their positions in this photo. The x and y coordinates are percentages
314   *         from the left and top edges of the photo, respectively.
315   */
316  public List<Tag> getTags() {
317    return unmodifiableList(tags);
318  }
319
320  /**
321   * All of the comments on this photo.
322   * 
323   * @return All of the comments on this photo.
324   * @since 1.6.5
325   */
326  public List<Comment> getComments() {
327    return unmodifiableList(comments);
328  }
329
330  /**
331   * Users who like the photo.
332   * 
333   * @return Users who like the photo.
334   * @since 1.6.5
335   */
336  public List<NamedFacebookType> getLikes() {
337    return unmodifiableList(likes);
338  }
339
340  /**
341   * The 4 different stored representations of the photo.
342   * 
343   * @return The 4 different stored representations of the photo.
344   * @since 1.6.5
345   */
346  public List<Image> getImages() {
347    return unmodifiableList(images);
348  }
349}
