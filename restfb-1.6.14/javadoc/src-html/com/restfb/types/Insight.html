023package com.restfb.types;
024
025import java.util.ArrayList;
026import java.util.List;
027
028import com.restfb.Facebook;
029import com.restfb.json.JsonObject;
030
031/**
032 * Represents the <a href="http://developers.facebook.com/docs/reference/api/insights" >Insight Graph API type</a>.
033 * 
034 * @author <a href="http://restfb.com">Mark Allen</a>
035 * @since 1.6.3
036 */
037public class Insight extends NamedFacebookType {
038  @Facebook
039  private String period;
040
041  @Facebook
042  private String description;
043
044  @Facebook
045  private List<JsonObject> values = new ArrayList<JsonObject>();
046
047  private static final long serialVersionUID = 1L;
048
049  /**
050   * Length of the period during which the insights were collected, e.g. 'day', 'week' or 'month'.
051   * 
052   * @return Length of the period during which the insights were collected.
053   */
054  public String getPeriod() {
055    return period;
056  }
057
058  /**
059   * The human-readable description of this Insight data.
060   * 
061   * @return The human-readable description of this Insight data.
062   */
063  public String getDescription() {
064    return description;
065  }
066
067  /**
068   * Data for this Insight as a list of <tt>{@link com.restfb.json.JsonObject}</tt> because its structure can vary
069   * depending on which type of Insight you're looking at.
070   * 
071   * @return Data for this Insight.
072   */
073  public List<JsonObject> getValues() {
074    return values;
075  }
076}
