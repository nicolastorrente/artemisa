023package com.restfb.types;
024
025import com.restfb.Facebook;
026
027/**
028 * Represents the <a href="https://developers.facebook.com/docs/test_users">Test User type</a>.
029 * 
030 * @author <a href="http://ex-nerd.com">Chris Petersen</a>
031 * @author <a href="http://restfb.com">Mark Allen</a>
032 * @since 1.6.10
033 */
034public class TestUser extends FacebookType {
035  @Facebook("access_token")
036  private String accessToken;
037
038  @Facebook("login_url")
039  private String loginUrl;
040
041  @Facebook
042  private String email;
043
044  @Facebook
045  private String password;
046
047  private static final long serialVersionUID = 1L;
048
049  /**
050   * You can use this access token to make API calls on behalf of the test user. This is available only if your app has
051   * been installed by the test user.
052   * 
053   * @return The test user's access token.
054   */
055  public String getAccessToken() {
056    return accessToken;
057  }
058
059  /**
060   * You can login as the test user by going to this url.
061   * 
062   * @return The test user's login url.
063   */
064  public String getLoginUrl() {
065    return loginUrl;
066  }
067
068  /**
069   * If logging in manually (that is, not using the login_url), you can use this as the user's email address.
070   * 
071   * @return The test user's email address.
072   */
073  public String getEmail() {
074    return email;
075  }
076
077  /**
078   * If logging in manually (that is, not using the login_url), you can use this as the user's password.
079   * 
080   * @return The test user's password.
081   */
082  public String getPassword() {
083    return password;
084  }
085}
