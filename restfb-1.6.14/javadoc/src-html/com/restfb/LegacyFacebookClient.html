023package com.restfb;
024
025import java.util.List;
026import java.util.Map;
027
028import com.restfb.exception.FacebookException;
029
030/**
031 * Specifies how a <a href="http://developers.facebook.com/docs/reference/rest/">Legacy Facebook API</a> client must
032 * operate.
033 * <p>
034 * Implementors must support Legacy authentication (API Key, Application Secret, and Session Key) as well as the <a
035 * href="http://developers.facebook.com/docs/guides/upgrade#oauth">new OAuth authentication</a> method. Using OAuth
036 * authentication is much simpler and strongly recommended.
037 * <p>
038 * Green-field projects should use the new <a href="http://developers.facebook.com/docs/api">Facebook Graph API</a> via
039 * {@link FacebookClient} instead.
040 * <p>
041 * <ul>
042 * <li>The {@link #execute(String, Parameter...)} family of methods should be used when performing API calls that return
043 * a single object.</li>
044 * <li>The {@link #executeForList(String, Class, Parameter...)} family of methods should be used when performing API
045 * calls that return a list of objects.</li>
046 * <li>The {@link #executeMultiquery(Map, Class, Parameter...)} family of methods should be used when performing <a
047 * href="http://wiki.developers.facebook.com/index.php/Fql.multiquery"> {@code fql.multiquery}</a> API calls.</li>
048 * </ul>
049 * 
050 * @author <a href="http://restfb.com">Mark Allen</a>
051 */
052public interface LegacyFacebookClient {
053  /**
054   * Executes a Facebook API method with the given {@code parameters}, ignoring the response.
055   * 
056   * @param method
057   *          The Facebook API method to call, e.g. {@code fql.query}.
058   * @param parameters
059   *          Parameters to include in the API call.
060   * @throws FacebookException
061   *           If an error occurs while performing the API call.
062   */
063  void execute(String method, Parameter... parameters);
064
065  /**
066   * Executes a Facebook API method with the given {@code parameters}, ignoring the response.
067   * 
068   * @param method
069   *          The Facebook API method to call, e.g. {@code fql.query}.
070   * @param sessionKey
071   *          A Facebook API session key if you're using the legacy API key/Secret key authentication scheme. Must be
072   *          {@code null} if using OAuth access token authentication.
073   * @param parameters
074   *          Parameters to include in the API call.
075   * @throws FacebookException
076   *           If an error occurs while performing the API call.
077   * @throws IllegalArgumentException
078   *           If {@code sessionKey} is provided when using OAuth access token authentication.
079   * @deprecated Use {@link #execute(String, Parameter...)} instead. Facebook is moving to OAuth and will stop
080   *             supporting the old session key authentication scheme soon.
081   */
082  @Deprecated
083  void execute(String method, String sessionKey, Parameter... parameters);
084
085  /**
086   * Executes a Facebook API method with the given {@code parameters}, mapping the API response to a single instance of
087   * type {@code resultType}.
088   * 
089   * @param <T>
090   *          Java type to map to.
091   * @param method
092   *          The Facebook API method to call, e.g. {@code fql.query}.
093   * @param resultType
094   *          Result type token.
095   * @param parameters
096   *          Parameters to include in the API call.
097   * @return An instance of type {@code resultType} which contains API response data.
098   * @throws FacebookException
099   *           If an error occurs while performing the API call.
100   */
101  <T> T execute(String method, Class<T> resultType, Parameter... parameters);
102
103  /**
104   * Executes a Facebook API method with the given {@code parameters}, mapping the API response to a single instance of
105   * type {@code resultType}.
106   * 
107   * @param <T>
108   *          Java type to map to.
109   * @param method
110   *          The Facebook API method to call, e.g. {@code fql.query}.
111   * @param sessionKey
112   *          A Facebook API session key if you're using the legacy API key/Secret key authentication scheme. Must be
113   *          {@code null} if using OAuth access token authentication.
114   * @param resultType
115   *          Result type token.
116   * @param parameters
117   *          Parameters to include in the API call.
118   * @return An instance of type {@code resultType} which contains API response data.
119   * @throws FacebookException
120   *           If an error occurs while performing the API call.
121   * @throws IllegalArgumentException
122   *           If {@code sessionKey} is provided when using OAuth access token authentication.
123   * @deprecated Use {@link #execute(String, Class, Parameter...)} instead. Facebook is moving to OAuth and will stop
124   *             supporting the old session key authentication scheme soon.
125   */
126  @Deprecated
127  <T> T execute(String method, String sessionKey, Class<T> resultType, Parameter... parameters);
128
129  /**
130   * Executes a Facebook API method with the given {@code parameters}, mapping the API response to a {@code List} of
131   * instances of type {@code resultType}.
132   * 
133   * @param <T>
134   *          Java type to map to.
135   * @param method
136   *          The Facebook API method to call, e.g. {@code fql.query}.
137   * @param resultType
138   *          Result type token.
139   * @param parameters
140   *          Parameters to include in the API call.
141   * @return A {@code List} of instances of type {@code resultType} which contain API response data.
142   * @throws FacebookException
143   *           If an error occurs while performing the API call.
144   */
145  <T> List<T> executeForList(String method, Class<T> resultType, Parameter... parameters);
146
147  /**
148   * Executes a Facebook API method with the given {@code parameters}, mapping the API response to a {@code List} of
149   * instances of type {@code resultType}.
150   * 
151   * @param <T>
152   *          Java type to map to.
153   * @param method
154   *          The Facebook API method to call, e.g. {@code fql.query}.
155   * @param sessionKey
156   *          A Facebook API session key if you're using the legacy API key/Secret key authentication scheme. Must be
157   *          {@code null} if using OAuth access token authentication.
158   * @param resultType
159   *          Result type token.
160   * @param parameters
161   *          Parameters to include in the API call.
162   * @return A {@code List} of instances of type {@code resultType} which contain API response data.
163   * @throws FacebookException
164   *           If an error occurs while performing the API call.
165   * @throws IllegalArgumentException
166   *           If {@code sessionKey} is provided when using OAuth access token authentication.
167   * @deprecated Use {@link #executeForList(String, Class, Parameter...)} instead. Facebook is moving to OAuth and will
168   *             stop supporting the old session key authentication scheme soon.
169   */
170  @Deprecated
171  <T> List<T> executeForList(String method, String sessionKey, Class<T> resultType, Parameter... parameters);
172
173  /**
174   * Executes the <a href="http://wiki.developers.facebook.com/index.php/Fql.multiquery"> {@code fql.multiquery}</a> API
175   * call, mapping the API response to a single instance of type {@code resultType}.
176   * <p>
177   * This method exists because the standard {@link #execute(String, Parameter...)} and
178   * {@link #executeForList(String, Class, Parameter...)} family of methods are not expressive enough to handle
179   * {@code fql.multiquery} in a non-verbose way.
180   * 
181   * @param <T>
182   *          Java type to map to.
183   * 
184   * @param queries
185   *          A mapping of query names to queries. This is marshaled to JSON and sent over the wire to the Facebook API
186   *          endpoint as the {@code queries} parameter.
187   * @param resultType
188   *          Result type token.
189   * @param additionalParameters
190   *          Additional parameters to include in the API call.
191   * @return An instance of type {@code resultType} which contains API response data.
192   * @throws FacebookException
193   *           If an error occurs while performing the API call.
194   * @since 1.1
195   */
196  <T> T executeMultiquery(Map<String, String> queries, Class<T> resultType, Parameter... additionalParameters);
197
198  /**
199   * Executes the <a href="http://wiki.developers.facebook.com/index.php/Fql.multiquery"> {@code fql.multiquery}</a> API
200   * call, mapping the API response to a single instance of type {@code resultType}.
201   * <p>
202   * This method exists because the standard {@link #execute(String, Parameter...)} and
203   * {@link #executeForList(String, Class, Parameter...)} family of methods are not expressive enough to handle
204   * {@code fql.multiquery} in a non-verbose way.
205   * 
206   * @param <T>
207   *          Java type to map to.
208   * @param queries
209   *          A mapping of query names to queries. This is marshaled to JSON and sent over the wire to the Facebook API
210   *          endpoint as the {@code queries} parameter.
211   * @param sessionKey
212   *          A Facebook API session key if you're using the legacy API key/Secret key authentication scheme. Must be
213   *          {@code null} if using OAuth access token authentication.
214   * @param resultType
215   *          Result type token.
216   * @param additionalParameters
217   *          Additional parameters to include in the API call.
218   * @return An instance of type {@code resultType} which contains API response data.
219   * @throws FacebookException
220   *           If an error occurs while performing the API call.
221   * @throws IllegalArgumentException
222   *           If {@code sessionKey} is provided when using OAuth access token authentication.
223   * @since 1.1
224   * @deprecated Use {@link #executeMultiquery(Map, Class, Parameter...)} instead. Facebook is moving to OAuth and will
225   *             stop supporting the old session key authentication scheme soon.
226   */
227  @Deprecated
228  <T> T executeMultiquery(Map<String, String> queries, String sessionKey, Class<T> resultType,
229      Parameter... additionalParameters);
230}
