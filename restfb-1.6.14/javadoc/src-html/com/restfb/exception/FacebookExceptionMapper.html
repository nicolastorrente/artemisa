023package com.restfb.exception;
024
025/**
026 * Specifies a method for mapping Graph and Old REST API exceptions to corresponding instances of
027 * {@code FacebookException}.
028 * 
029 * @author <a href="http://restfb.com">Mark Allen</a>
030 * @since 1.6
031 */
032public interface FacebookExceptionMapper {
033  /**
034   * Given a Facebook API exception type and message, generates an instance of the corresponding
035   * {@code FacebookGraphException} or one of its subclasses.
036   * 
037   * @param errorCode
038   *          Old REST API exception error code field, e.g. 190.
039   * @param errorSubcode
040   *          Old REST API exception error subcode field, e.g. 459.
041   * @param httpStatusCode
042   *          The HTTP status code returned by the server, e.g. 500.
043   * @param type
044   *          Graph API exception type field, e.g. "OAuthException".
045   * @param message
046   *          Graph or Old REST API message field, e.g. "Invalid access token signature."
047   * @return An appropriate {@code FacebookException} subclass.
048   */
049  FacebookException exceptionForTypeAndMessage(Integer errorCode, Integer errorSubcode, Integer httpStatusCode, String type, String message);
050}
