023package com.restfb.exception;
024
025import static java.lang.String.format;
026
027/**
028 * Indicates that the Legacy REST Facebook endpoint returned JSON which indicates an error condition.
029 * <p>
030 * This exception may also be thrown when executing certain operations against the Graph API, e.g. FQL queries or Batch
031 * API calls.
032 * <p>
033 * Example: <code>{"error_code": 2, "error_msg": "The service is not available at this time.", ...}</code>.
034 * 
035 * @author <a href="http://restfb.com">Mark Allen</a>
036 */
037public class FacebookResponseStatusException extends FacebookException {
038  /**
039   * The Facebook API error code.
040   */
041  private Integer errorCode;
042
043  /**
044   * The Facebook API error message.
045   */
046  private String errorMessage;
047
048  private static final long serialVersionUID = 1L;
049
050  /**
051   * Creates an exception with the given message and error code.
052   * 
053   * @param errorCode
054   *          Value of the Facebook response attribute {@code error_code}.
055   * @param errorMessage
056   *          Value of the Facebook response attribute {@code error_msg}.
057   */
058  public FacebookResponseStatusException(Integer errorCode, String errorMessage) {
059    super(format("Received Facebook error response (code %d): %s", errorCode, errorMessage));
060    this.errorCode = errorCode;
061    this.errorMessage = errorMessage;
062  }
063
064  /**
065   * Gets the Facebook API error code.
066   * 
067   * @return The Facebook API error code.
068   */
069  public Integer getErrorCode() {
070    return errorCode;
071  }
072
073  /**
074   * Gets the Facebook API error message.
075   * 
076   * @return The Facebook API error message.
077   */
078  public String getErrorMessage() {
079    return errorMessage;
080  }
081}
