023package com.restfb.exception;
024
025import static java.lang.String.format;
026
027/**
028 * Indicates that the Facebook Graph API endpoint returned JSON which indicates an error condition.
029 * <p>
030 * Example:<code>
031  {
032      "error": {
033        "type": "Exception",
034        "message": "...",
035        "code": 210,
036        "error_subcode": 123        
037      }
038  } </code>
039 * 
040 * @author <a href="http://restfb.com">Mark Allen</a>
041 * @since 1.5
042 */
043public class FacebookGraphException extends FacebookException {
044  private static final long serialVersionUID = 1L;
045
046  /**
047   * The Facebook Graph API error type.
048   */
049  private String errorType;
050
051  /**
052   * The Facebook API error message.
053   */
054  private String errorMessage;
055
056  /**
057   * The Facebook API error code.
058   */
059  private Integer errorCode;
060
061  /**
062   * The Facebook API error subcode.
063   */
064  private Integer errorSubcode;
065
066  /**
067   * The HTTP status code returned by the server.
068   */
069  private Integer httpStatusCode;
070
071  /**
072   * Creates an exception with the given error type and message.
073   * 
074   * @param errorType
075   *          Value of the Facebook response attribute {@code error.type}.
076   * @param errorMessage
077   *          Value of the Facebook response attribute {@code error.message}.
078   * @param errorCode
079   *          Value of the Facebook response attribute {@code error.code}.
080   * @param errorSubcode
081   *          Value of the Facebook response attribute {@code error.error_subcode}.
082   * @param httpStatusCode
083   *          The HTTP status code returned by the server, e.g. 500.
084   */
085  public FacebookGraphException(String errorType, String errorMessage, Integer errorCode, Integer errorSubcode,
086      Integer httpStatusCode) {
087    super(format("Received Facebook error response of type %s: %s (code %s, subcode %s)", errorType, errorMessage,
088      errorCode, errorSubcode));
089    this.errorType = errorType;
090    this.errorMessage = errorMessage;
091    this.errorCode = errorCode;
092    this.errorSubcode = errorSubcode;
093    this.httpStatusCode = httpStatusCode;
094  }
095
096  /**
097   * Gets the Facebook Graph API error type.
098   * 
099   * @return The Facebook Graph API error type.
100   */
101  public String getErrorType() {
102    return errorType;
103  }
104
105  /**
106   * Gets the Facebook Graph API error message.
107   * 
108   * @return The Facebook Graph API error message.
109   */
110  public String getErrorMessage() {
111    return errorMessage;
112  }
113
114  /**
115   * Gets the Facebook API error code.
116   * 
117   * @return The Facebook API error code.
118   */
119  public Integer getErrorCode() {
120    return errorCode;
121  }
122
123  /**
124   * Gets the Facebook API error subcode.
125   * 
126   * @return The Facebook API error subcode.
127   */
128  public Integer getErrorSubcode() {
129    return errorSubcode;
130  }
131
132  /**
133   * Gets the HTTP status code returned by the server.
134   * 
135   * @return The HTTP status code returned by the server.
136   * @since 1.6.10
137   */
138  public Integer getHttpStatusCode() {
139    return httpStatusCode;
140  }
141}
