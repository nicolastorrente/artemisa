023package com.restfb;
024
025import static com.restfb.util.DateUtils.FACEBOOK_LONG_DATE_FORMAT;
026import static com.restfb.util.StringUtils.isBlank;
027import static com.restfb.util.StringUtils.trimToEmpty;
028import static java.lang.String.format;
029
030import java.text.SimpleDateFormat;
031import java.util.Date;
032
033import com.restfb.exception.FacebookJsonMappingException;
034
035/**
036 * Representation of a Facebook API request parameter.
037 * 
038 * @author <a href="http://restfb.com">Mark Allen</a>
039 */
040public final class Parameter {
041  /**
042   * Parameter name.
043   */
044  public final String name;
045
046  /**
047   * Parameter value.
048   */
049  public final String value;
050
051  /**
052   * Creates a new parameter with the given {@code name} and {@code value}.
053   * 
054   * @param name
055   *          The parameter name.
056   * @param value
057   *          The parameter value.
058   * @param jsonMapper
059   *          Mapper for converting the parameter value to JSON.
060   * @throws IllegalArgumentException
061   *           If {@code name} is {@code null} or a blank string or either {@code value} or {@code jsonMapper} is
062   *           {@code null}.
063   */
064  private Parameter(String name, Object value, JsonMapper jsonMapper) {
065    if (isBlank(name) || value == null)
066      throw new IllegalArgumentException(Parameter.class + " instances must have a non-blank name and non-null value.");
067
068    if (jsonMapper == null)
069      throw new IllegalArgumentException("Provided " + JsonMapper.class + " must not be null.");
070
071    this.name = trimToEmpty(name).toLowerCase();
072
073    // Special handling for Date types - turn them into Facebook date strings.
074    // Otherwise, use the JSON value of the type.
075    this.value =
076        value instanceof Date ? new SimpleDateFormat(FACEBOOK_LONG_DATE_FORMAT).format(value) : jsonMapper
077          .toJson(value);
078  }
079
080  /**
081   * Factory method which provides an instance with the given {@code name} and {@code value}.
082   * <p>
083   * The {@code value} parameter is often a {@link String} or primitive type like {@link Integer}, but you may pass in a
084   * {@link java.util.List}, {@link java.util.Map}, or your own <tt>@Facebook</tt>-annotated Javabean and it will be
085   * converted to JSON automatically. See the "attachment" section of <a
086   * href="http://wiki.developers.facebook.com/index.php/Stream.publish">the stream.publish API documentation</a> for an
087   * example of where this is useful.
088   * 
089   * @param name
090   *          The parameter name.
091   * @param value
092   *          The parameter value.
093   * @return A {@code Parameter} instance with the given {@code name} and {@code value}.
094   * @throws IllegalArgumentException
095   *           If {@code name} or {@code value} is {@code null} or a blank string.
096   * @throws FacebookJsonMappingException
097   *           If an error occurs when converting {@code value} to JSON.
098   */
099  public static Parameter with(String name, Object value) {
100    return Parameter.with(name, value, new DefaultJsonMapper());
101  }
102
103  /**
104   * Factory method which provides an instance with the given {@code name} and {@code value}, using the provided
105   * {@code jsonMapper} to turn {@code value} into a JSON string.
106   * <p>
107   * The {@code value} parameter is often a {@link String} or primitive type like {@link Integer}, but you may pass in a
108   * {@link java.util.List}, {@link java.util.Map}, or your own <tt>@Facebook</tt>-annotated Javabean and it will be
109   * converted to JSON automatically. See the "attachment" section of <a
110   * href="http://wiki.developers.facebook.com/index.php/Stream.publish">the stream.publish API documentation</a> for an
111   * example of where this is useful.
112   * 
113   * @param name
114   *          The parameter name.
115   * @param value
116   *          The parameter value.
117   * @return A {@code Parameter} instance with the given {@code name} and {@code value}.
118   * @throws IllegalArgumentException
119   *           If {@code name} or {@code value} is {@code null} or a blank string.
120   * @throws FacebookJsonMappingException
121   *           If an error occurs when converting {@code value} to JSON.
122   */
123  public static Parameter with(String name, Object value, JsonMapper jsonMapper) {
124    return new Parameter(name, value, jsonMapper);
125  }
126
127  /**
128   * @see java.lang.Object#equals(java.lang.Object)
129   */
130  @Override
131  public boolean equals(Object obj) {
132    if (obj == null)
133      return false;
134    if (!getClass().equals(obj.getClass()))
135      return false;
136
137    Parameter other = (Parameter) obj;
138
139    if (this.name != other.name && (!this.name.equals(other.name)))
140      return false;
141    if (this.value != other.value && (!this.value.equals(other.value)))
142      return false;
143
144    return true;
145  }
146
147  /**
148   * @see java.lang.Object#hashCode()
149   */
150  @Override
151  public int hashCode() {
152    int hash = 7;
153    hash = 37 * hash + this.name.hashCode();
154    hash = 41 * hash + this.value.hashCode();
155    return hash;
156  }
157
158  /**
159   * @see java.lang.Object#toString()
160   */
161  @Override
162  public String toString() {
163    return format("Parameter[%s=%s]", name, value);
164  }
165}
