001package com.restfb.json;
002
003/**
004 * The <code>JsonString</code> interface allows a <code>toJSONString()</code> method so that a class can change the
005 * behavior of <code>JsonObject.toString()</code>, <code>JsonArray.toString()</code>, and <code>JsonWriter.value(</code>
006 * Object<code>)</code>. The <code>toJSONString</code> method will be used instead of the default behavior of using the
007 * Object's <code>toString()</code> method and quoting the result.
008 */
009public interface JsonString {
010  /**
011   * The <code>toJSONString</code> method allows a class to produce its own JSON serialization.
012   * 
013   * @return A strictly syntactically correct JSON text.
014   */
015  public String toJsonString();
016}
