253  /**
254   * @see com.restfb.LegacyFacebookClient#executeMultiquery(java.util.Map, java.lang.String, java.lang.Class,
255   *      com.restfb.Parameter[])
256   */
257  @Override
258  public <T> T executeMultiquery(Map<String, String> queries, String sessionKey, Class<T> resultType,
259      Parameter... additionalParameters) {
260    List<Parameter> parameters = new ArrayList<Parameter>();
261    parameters.add(Parameter.with("queries", queriesToJson(queries)));
262
263    for (Parameter additionalParameter : additionalParameters) {
264      if (additionalParameter.name.equals("queries"))
265        throw new IllegalArgumentException("You cannot specify a parameter named 'queries' "
266            + "because it's reserved for use by RestFB for this call. "
267            + "Specify your queries in the Map that gets passed to this method.");
268
269      parameters.add(additionalParameter);
270    }
271
272    JsonObject normalizedJson = new JsonObject();
273
274    try {
275      JsonArray jsonArray =
276          new JsonArray(makeRequest("fql.multiquery", sessionKey, parameters.toArray(new Parameter[0])));
277
278      for (int i = 0; i < jsonArray.length(); i++) {
279        JsonObject jsonObject = jsonArray.getJsonObject(i);
280
281        // For empty resultsets, Facebook will return an empty object instead of
282        // an empty list. Hack around that here.
283        JsonArray resultsArray =
284            jsonObject.get("fql_result_set") instanceof JsonArray ? jsonObject.getJsonArray("fql_result_set")
285                : new JsonArray();
286
287        normalizedJson.put(jsonObject.getString("name"), resultsArray);
288      }
289    } catch (JsonException e) {
290      throw new FacebookJsonMappingException("Unable to process fql.multiquery JSON response", e);
291    }
292
293    return jsonMapper.toJavaObject(normalizedJson.toString(), resultType);
294  }

296  /**
297   * Coordinates the process of verifying and transforming API parameters, executing the API POST, and processing the
298   * response we receive from the endpoint.
299   * 
300   * @param method
301   *          Facebook API method name.
302   * @param sessionKey
303   *          Facebook API session key (can be {@code null} or a blank string).
304   * @param parameters
305   *          Arbitrary number of parameters to send along to Facebook as part of the API call.
306   * @return The JSON returned by Facebook for the API call.
307   * @throws FacebookException
308   *           If an error occurs while making the Facebook API POST or processing the response.
309   */
310  protected String makeRequest(String method, String sessionKey, Parameter... parameters) {
311    // Make sure we're not provided with any params that conflict with what
312    // we're passing to FB internally
313    verifyParameterLegality(parameters);
314
315    // Turn the set of parameters into a string per the API spec
316    String parametersAsString = toParameterString(method, sessionKey, parameters);
317
318    Response response = null;
319
320    // Perform a POST to the API endpoint
321    try {
322      response = webRequestor.executePost(createEndpointForApiCall(method, false), parametersAsString);
323    } catch (Throwable t) {
324      throw new FacebookNetworkException("Facebook POST failed", t);
325    }
326
327    // If we get any HTTP response code other than a 200 OK, throw an exception
328    if (HTTP_OK != response.getStatusCode())
329      throw new FacebookNetworkException("Facebook POST failed", response.getStatusCode());
330
331    String json = response.getBody();
332
333    // If the response contained an error code, throw an exception
334    throwLegacyFacebookResponseStatusExceptionIfNecessary(json, response.getStatusCode());
335
336    return json;
337  }

339  /**
340   * Given basic request information, generate the parameter string to be included in the Facebook API POST.
341   * 
342   * @param method
343   *          Facebook API method name.
344   * @param sessionKey
345   *          Facebook API session key (can be {@code null} or a blank string).
346   * @param parameters
347   *          Arbitrary number of extra parameters to include in the request.
348   * @return The parameter string to include in the Facebook API POST.
349   * @throws IllegalArgumentException
350   *           If a session key is provided but we're using OAuth authentication instead.
351   */
352  protected String toParameterString(String method, String sessionKey, Parameter... parameters) {
353    Map<String, String> sortedParameters = new TreeMap<String, String>();
354
355    for (Parameter param : parameters)
356      sortedParameters.put(param.name, param.value);
357
358    sortedParameters.put(FORMAT_PARAM_NAME, FORMAT_PARAM_VALUE);
359    sortedParameters.put(METHOD_PARAM_NAME, method);
360
361    // New OAuth access token parameter is simple: just one value.
362    // The legacy stuff requires a bunch more work.
363    if (usesAccessTokenAuthentication()) {
364      if (sessionKey != null)
365        throw new IllegalArgumentException("If you're using the OAuth access token "
366            + "for authentication, you cannot " + "specify a session key.");
367
368      sortedParameters.put(ACCESS_TOKEN_PARAM_NAME, accessToken);
369    } else {
370      sortedParameters.put(API_KEY_PARAM_NAME, apiKey);
371      sortedParameters.put(VERSION_PARAM_NAME, VERSION_PARAM_VALUE);
372      sortedParameters.put(CALL_ID_PARAM_NAME, String.valueOf(System.currentTimeMillis()));
373
374      if (!isBlank(sessionKey))
375        sortedParameters.put(SESSION_KEY_PARAM_NAME, sessionKey);
376
377      sortedParameters.put(SIG_PARAM_NAME, generateSignature(sortedParameters));
378    }
379
380    StringBuilder parameterStringBuilder = new StringBuilder();
381    boolean first = true;
382
383    for (Entry<String, String> entry : sortedParameters.entrySet()) {
384      if (first)
385        first = false;
386      else
387        parameterStringBuilder.append("&");
388
389      parameterStringBuilder.append(urlEncode(entry.getKey()));
390      parameterStringBuilder.append("=");
391      parameterStringBuilder.append(usesAccessTokenAuthentication() ? urlEncodedValueForParameterName(entry.getKey(),
392        entry.getValue()) : urlEncode(entry.getValue()));
393    }
394
395    return parameterStringBuilder.toString();
396  }

398  /**
399   * Given a sorted map of parameter names to values, calculate and return the Facebook API signature as defined by
400   * http://wiki.developers.facebook.com/index.php/Verifying_The_Signature.
401   * 
402   * @param sortedParameters
403   *          Parameter name/value mappings, sorted alphabetically.
404   * @return The Facebook API signature which matches the given parameter map.
405   */
406  protected String generateSignature(Map<String, String> sortedParameters) {
407    StringBuilder parameterString = new StringBuilder();
408
409    for (Entry<String, String> entry : sortedParameters.entrySet()) {
410      parameterString.append(entry.getKey());
411      parameterString.append("=");
412      parameterString.append(entry.getValue());
413    }
414
415    parameterString.append(secretKey);
416    return generateMd5(parameterString.toString());
417  }

419  /**
420   * Generate an MD5 hash of the given {@code string}.
421   * 
422   * @param string
423   *          The string for which an MD5 hash is calculated.
424   * @return The MD5 hash of the given {@code string}.
425   * @throws IllegalStateException
426   *           If MD5 hashing isn't supported on this platform (should never occur).
427   */
428  protected String generateMd5(String string) {
429    try {
430      MessageDigest messageDigest = MessageDigest.getInstance("MD5");
431      byte[] bytes = toBytes(string);
432
433      StringBuilder result = new StringBuilder();
434      for (byte b : messageDigest.digest(bytes)) {
435        result.append(Integer.toHexString((b & 0xf0) >>> 4));
436        result.append(Integer.toHexString(b & 0x0f));
437      }
438      return result.toString();
439    } catch (NoSuchAlgorithmException e) {
440      // Should never happen
441      throw new IllegalStateException("MD5 isn't available on this JVM", e);
442    }
443  }
