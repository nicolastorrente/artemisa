274  /**
275   * @see com.restfb.FacebookClient#fetchConnectionPage(java.lang.String, java.lang.Class)
276   */
277  public <T> Connection<T> fetchConnectionPage(final String connectionPageUrl, Class<T> connectionType) {
278    String connectionJson = null;
279    if (!isBlank(accessToken) && !isBlank(appSecret)) {
280      connectionJson = makeRequestAndProcessResponse(new Requestor() {
281        public Response makeRequest() throws IOException {
282          return webRequestor.executeGet(String.format("%s&%s=%s", connectionPageUrl,
283            urlEncode(APP_SECRET_PROOF_PARAM_NAME), obtainAppSecretProof(accessToken, appSecret)));
284        }
285      });
286    } else {
287      connectionJson = makeRequestAndProcessResponse(new Requestor() {
288        /**
289         * @see com.restfb.DefaultFacebookClient.Requestor#makeRequest()
290         */
291        public Response makeRequest() throws IOException {
292          return webRequestor.executeGet(connectionPageUrl);
293        }
294      });
295    }
296
297    return new Connection<T>(this, connectionJson, connectionType);
298  }

309  /**
310   * @see com.restfb.FacebookClient#fetchObjects(java.util.List, java.lang.Class, com.restfb.Parameter[])
311   */
312  @SuppressWarnings("unchecked")
313  public <T> T fetchObjects(List<String> ids, Class<T> objectType, Parameter... parameters) {
314    verifyParameterPresence("ids", ids);
315    verifyParameterPresence("connectionType", objectType);
316
317    if (ids.size() == 0)
318      throw new IllegalArgumentException("The list of IDs cannot be empty.");
319
320    for (Parameter parameter : parameters)
321      if (IDS_PARAM_NAME.equals(parameter.name))
322        throw new IllegalArgumentException("You cannot specify the '" + IDS_PARAM_NAME + "' URL parameter yourself - "
323            + "RestFB will populate this for you with " + "the list of IDs you passed to this method.");
324
325    // Normalize the IDs
326    for (int i = 0; i < ids.size(); i++) {
327      String id = ids.get(i).trim().toLowerCase();
328      if ("".equals(id))
329        throw new IllegalArgumentException("The list of IDs cannot contain blank strings.");
330      ids.set(i, id);
331    }
332
333    try {
334      JsonObject jsonObject =
335          new JsonObject(makeRequest("",
336            parametersWithAdditionalParameter(Parameter.with(IDS_PARAM_NAME, join(ids)), parameters)));
337
338      return objectType.equals(JsonObject.class) ? (T) jsonObject : jsonMapper.toJavaObject(jsonObject.toString(),
339        objectType);
340    } catch (JsonException e) {
341      throw new FacebookJsonMappingException("Unable to map connection JSON to Java objects", e);
342    }
343  }

368   * @see com.restfb.FacebookClient#executeMultiquery(java.util.Map, java.lang.Class, com.restfb.Parameter[])
369   */
370  @SuppressWarnings("unchecked")
371  public <T> T executeMultiquery(Map<String, String> queries, Class<T> objectType, Parameter... parameters) {
372    verifyParameterPresence("objectType", objectType);
373
374    for (Parameter parameter : parameters)
375      if (QUERIES_PARAM_NAME.equals(parameter.name))
376        throw new IllegalArgumentException("You cannot specify the '" + QUERIES_PARAM_NAME
377            + "' URL parameter yourself - " + "RestFB will populate this for you with "
378            + "the queries you passed to this method.");
379
380    try {
381      JsonArray jsonArray =
382          new JsonArray(makeRequest("fql.multiquery", false, false, null,
383            parametersWithAdditionalParameter(Parameter.with(QUERIES_PARAM_NAME, queriesToJson(queries)), parameters)));
384
385      JsonObject normalizedJson = new JsonObject();
386
387      for (int i = 0; i < jsonArray.length(); i++) {
388        JsonObject jsonObject = jsonArray.getJsonObject(i);
389
390        // For empty resultsets, Facebook will return an empty object instead of
391        // an empty list. Hack around that here.
392        JsonArray resultsArray =
393            jsonObject.get("fql_result_set") instanceof JsonArray ? jsonObject.getJsonArray("fql_result_set")
394                : new JsonArray();
395
396        normalizedJson.put(jsonObject.getString("name"), resultsArray);
397      }
398
399      return objectType.equals(JsonObject.class) ? (T) normalizedJson : jsonMapper.toJavaObject(
400        normalizedJson.toString(), objectType);
401    } catch (JsonException e) {
402      throw new FacebookJsonMappingException("Unable to process fql.multiquery JSON response", e);
403    }
404  }

598  /**
599   * @see com.restfb.FacebookClient#parseSignedRequest(java.lang.String, java.lang.String, java.lang.Class)
600   */
601  @Override
602  @SuppressWarnings("unchecked")
603  public <T> T parseSignedRequest(String signedRequest, String appSecret, Class<T> objectType) {
604    verifyParameterPresence("signedRequest", signedRequest);
605    verifyParameterPresence("appSecret", appSecret);
606    verifyParameterPresence("objectType", objectType);
607
608    String[] signedRequestTokens = signedRequest.split("[.]");
609
610    if (signedRequestTokens.length != 2)
611      throw new FacebookSignedRequestParsingException(format(
612        "Signed request '%s' is expected to be signature and payload strings separated by a '.'", signedRequest));
613
614    String encodedSignature = signedRequestTokens[0];
615    String urlDecodedSignature = urlDecodeSignedRequestToken(encodedSignature);
616    byte[] signature = decodeBase64(urlDecodedSignature);
617
618    String encodedPayload = signedRequestTokens[1];
619    String urlDecodedPayload = urlDecodeSignedRequestToken(encodedPayload);
620    String payload = StringUtils.toString(decodeBase64(urlDecodedPayload));
621
622    // Convert payload to a JsonObject so we can pull algorithm data out of it
623    JsonObject payloadObject = getJsonMapper().toJavaObject(payload, JsonObject.class);
624
625    if (!payloadObject.has("algorithm"))
626      throw new FacebookSignedRequestParsingException("Unable to detect algorithm used for signed request");
627
628    String algorithm = payloadObject.getString("algorithm");
629
630    if (!verifySignedRequest(appSecret, algorithm, encodedPayload, signature))
631      throw new FacebookSignedRequestVerificationException(
632        "Signed request verification failed. Are you sure the request was made for the app identified by the app secret you provided?");
633
634    // Marshal to the user's preferred type.
635    // If the user asked for a JsonObject, send back the one we already parsed.
636    return objectType.equals(JsonObject.class) ? (T) payloadObject : getJsonMapper().toJavaObject(payload, objectType);
637  }

651  /**
652   * Verifies that the signed request is really from Facebook.
653   * 
654   * @param appSecret
655   *          The secret for the app that can verify this signed request.
656   * @param algorithm
657   *          Signature algorithm specified by FB in the decoded payload.
658   * @param encodedPayload
659   *          The encoded payload used to generate a signature for comparison against the provided {@code signature}.
660   * @param signature
661   *          The decoded signature extracted from the signed request. Compared against a signature generated from
662   *          {@code encodedPayload}.
663   * @return {@code true} if the signed request is verified, {@code false} if not.
664   */
665  protected boolean verifySignedRequest(String appSecret, String algorithm, String encodedPayload, byte[] signature) {
666    verifyParameterPresence("appSecret", appSecret);
667    verifyParameterPresence("algorithm", algorithm);
668    verifyParameterPresence("encodedPayload", encodedPayload);
669    verifyParameterPresence("signature", signature);
670
671    // Normalize algorithm name...FB calls it differently than Java does
672    if ("HMAC-SHA256".equalsIgnoreCase(algorithm))
673      algorithm = "HMACSHA256";
674
675    try {
676      Mac mac = Mac.getInstance(algorithm);
677      mac.init(new SecretKeySpec(toBytes(appSecret), algorithm));
678      byte[] payloadSignature = mac.doFinal(toBytes(encodedPayload));
679      return Arrays.equals(signature, payloadSignature);
680    } catch (Exception e) {
681      throw new FacebookSignedRequestVerificationException("Unable to perform signed request verification", e);
682    }
683  }

732  /**
733   * Coordinates the process of executing the API request GET/POST and processing the response we receive from the
734   * endpoint.
735   * 
736   * @param endpoint
737   *          Facebook Graph API endpoint.
738   * @param executeAsPost
739   *          {@code true} to execute the web request as a {@code POST}, {@code false} to execute as a {@code GET}.
740   * @param executeAsDelete
741   *          {@code true} to add a special 'treat this request as a {@code DELETE}' parameter.
742   * @param binaryAttachment
743   *          A binary file to include in a {@code POST} request. Pass {@code null} if no attachment should be sent.
744   * @param parameters
745   *          Arbitrary number of parameters to send along to Facebook as part of the API call.
746   * @return The JSON returned by Facebook for the API call.
747   * @throws FacebookException
748   *           If an error occurs while making the Facebook API POST or processing the response.
749   */
750  protected String makeRequest(String endpoint, final boolean executeAsPost, boolean executeAsDelete,
751      final List<BinaryAttachment> binaryAttachments, Parameter... parameters) {
752    verifyParameterLegality(parameters);
753
754    if (executeAsDelete)
755      parameters = parametersWithAdditionalParameter(Parameter.with(METHOD_PARAM_NAME, "delete"), parameters);
756
757    trimToEmpty(endpoint).toLowerCase();
758    if (!endpoint.startsWith("/"))
759      endpoint = "/" + endpoint;
760
761    final String fullEndpoint =
762        createEndpointForApiCall(endpoint, binaryAttachments != null && binaryAttachments.size() > 0);
763    final String parameterString = toParameterString(parameters);
764
765    return makeRequestAndProcessResponse(new Requestor() {
766      /**
767       * @see com.restfb.DefaultFacebookClient.Requestor#makeRequest()
768       */
769      public Response makeRequest() throws IOException {
770        return executeAsPost ? webRequestor.executePost(fullEndpoint, parameterString, binaryAttachments == null ? null
771            : binaryAttachments.toArray(new BinaryAttachment[] {})) : webRequestor.executeGet(fullEndpoint + "?"
772            + parameterString);
773      }
774    });
775  }

777  /**
778   * @see com.restfb.FacebookClient#obtainAppSecretProof(java.lang.String, java.lang.String)
779   */
780  @Override
781  public String obtainAppSecretProof(String accessToken, String appSecret) {
782    verifyParameterPresence("accessToken", accessToken);
783    verifyParameterPresence("appSecret", appSecret);
784
785    try {
786      byte[] key = appSecret.getBytes();
787      SecretKeySpec signingKey = new SecretKeySpec(key, "HmacSHA256");
788      Mac mac = Mac.getInstance("HmacSHA256");
789      mac.init(signingKey);
790      byte[] raw = mac.doFinal(accessToken.getBytes());
791      byte[] hex = encodeHex(raw);
792      return new String(hex, "UTF-8");
793    } catch (Exception e) {
794      throw new IllegalStateException("Creation of appsecret_proof has failed", e);
795    }
796  }

798  protected static interface Requestor {
799    Response makeRequest() throws IOException;
800  }
801
802  protected String makeRequestAndProcessResponse(Requestor requestor) {
803    Response response = null;
804
805    // Perform a GET or POST to the API endpoint
806    try {
807      response = requestor.makeRequest();
808    } catch (Throwable t) {
809      throw new FacebookNetworkException("Facebook request failed", t);
810    }
811
812    // If we get any HTTP response code other than a 200 OK or 400 Bad Request
813    // or 401 Not Authorized or 403 Forbidden or 404 Not Found or 500 Internal
814    // Server Error,
815    // throw an exception.
816    if (HTTP_OK != response.getStatusCode() && HTTP_BAD_REQUEST != response.getStatusCode()
817        && HTTP_UNAUTHORIZED != response.getStatusCode() && HTTP_NOT_FOUND != response.getStatusCode()
818        && HTTP_INTERNAL_ERROR != response.getStatusCode() && HTTP_FORBIDDEN != response.getStatusCode())
819      throw new FacebookNetworkException("Facebook request failed", response.getStatusCode());
820
821    String json = response.getBody();
822
823    // If the response contained an error code, throw an exception.
824    throwFacebookResponseStatusExceptionIfNecessary(json, response.getStatusCode());
825
826    // If there was no response error information and this was a 500 or 401
827    // error, something weird happened on Facebook's end. Bail.
828    if (HTTP_INTERNAL_ERROR == response.getStatusCode() || HTTP_UNAUTHORIZED == response.getStatusCode())
829      throw new FacebookNetworkException("Facebook request failed", response.getStatusCode());
830
831    return json;
832  }

834  /**
835   * Throws an exception if Facebook returned an error response. Using the Graph API, it's possible to see both the new
836   * Graph API-style errors as well as Legacy API-style errors, so we have to handle both here. This method extracts
837   * relevant information from the error JSON and throws an exception which encapsulates it for end-user consumption.
838   * <p>
839   * For Graph API errors:
840   * <p>
841   * If the {@code error} JSON field is present, we've got a response status error for this API call.
842   * <p>
843   * For Legacy errors (e.g. FQL):
844   * <p>
845   * If the {@code error_code} JSON field is present, we've got a response status error for this API call.
846   * 
847   * @param json
848   *          The JSON returned by Facebook in response to an API call.
849   * @param httpStatusCode
850   *          The HTTP status code returned by the server, e.g. 500.
851   * @throws FacebookGraphException
852   *           If the JSON contains a Graph API error response.
853   * @throws FacebookResponseStatusException
854   *           If the JSON contains an Legacy API error response.
855   * @throws FacebookJsonMappingException
856   *           If an error occurs while processing the JSON.
857   */
858  protected void throwFacebookResponseStatusExceptionIfNecessary(String json, Integer httpStatusCode) {
859    // If we have a legacy exception, throw it.
860    throwLegacyFacebookResponseStatusExceptionIfNecessary(json, httpStatusCode);
861
862    // If we have a batch API exception, throw it.
863    throwBatchFacebookResponseStatusExceptionIfNecessary(json, httpStatusCode);
864
865    try {
866      // If the result is not an object, bail immediately.
867      if (!json.startsWith("{"))
868        return;
869
870      JsonObject errorObject = new JsonObject(json);
871
872      if (errorObject == null || !errorObject.has(ERROR_ATTRIBUTE_NAME))
873        return;
874
875      JsonObject innerErrorObject = errorObject.getJsonObject(ERROR_ATTRIBUTE_NAME);
876
877      // If there's an Integer error code, pluck it out.
878      Integer errorCode =
879          innerErrorObject.has(ERROR_CODE_ATTRIBUTE_NAME) ? toInteger(innerErrorObject
880            .getString(ERROR_CODE_ATTRIBUTE_NAME)) : null;
881      Integer errorSubcode =
882          innerErrorObject.has(ERROR_SUBCODE_ATTRIBUTE_NAME) ? toInteger(innerErrorObject
883            .getString(ERROR_SUBCODE_ATTRIBUTE_NAME)) : null;
884
885      throw graphFacebookExceptionMapper
886        .exceptionForTypeAndMessage(errorCode, errorSubcode, httpStatusCode,
887          innerErrorObject.getString(ERROR_TYPE_ATTRIBUTE_NAME),
888          innerErrorObject.getString(ERROR_MESSAGE_ATTRIBUTE_NAME));
889    } catch (JsonException e) {
890      throw new FacebookJsonMappingException("Unable to process the Facebook API response", e);
891    }
892  }

894  /**
895   * If the {@code error} and {@code error_description} JSON fields are present, we've got a response status error for
896   * this batch API call. Extracts relevant information from the JSON and throws an exception which encapsulates it for
897   * end-user consumption.
898   * 
899   * @param json
900   *          The JSON returned by Facebook in response to a batch API call.
901   * @param httpStatusCode
902   *          The HTTP status code returned by the server, e.g. 500.
903   * @throws FacebookResponseStatusException
904   *           If the JSON contains an error code.
905   * @throws FacebookJsonMappingException
906   *           If an error occurs while processing the JSON.
907   * @since 1.6.5
908   */
909  protected void throwBatchFacebookResponseStatusExceptionIfNecessary(String json, Integer httpStatusCode) {
910    try {
911      // If this is not an object, it's not an error response.
912      if (!json.startsWith("{"))
913        return;
914
915      JsonObject errorObject = null;
916
917      // We need to swallow exceptions here because it's possible to get a legit
918      // Facebook response that contains illegal JSON (e.g.
919      // users.getLoggedInUser returning 1240077) - we're only interested in
920      // whether or not there's an error_code field present.
921      try {
922        errorObject = new JsonObject(json);
923      } catch (JsonException e) {}
924
925      if (errorObject == null || !errorObject.has(BATCH_ERROR_ATTRIBUTE_NAME)
926          || !errorObject.has(BATCH_ERROR_DESCRIPTION_ATTRIBUTE_NAME))
927        return;
928
929      throw legacyFacebookExceptionMapper.exceptionForTypeAndMessage(errorObject.getInt(BATCH_ERROR_ATTRIBUTE_NAME),
930        null, httpStatusCode, null, errorObject.getString(BATCH_ERROR_DESCRIPTION_ATTRIBUTE_NAME));
931    } catch (JsonException e) {
932      throw new FacebookJsonMappingException("Unable to process the Facebook API response", e);
933    }
934  }

949   * A canned implementation of {@code FacebookExceptionMapper} that maps Graph API exceptions.
950   * <p>
951   * Thanks to BatchFB's Jeff Schnitzer for doing some of the legwork to find these exception type names.
952   * 
953   * @author <a href="http://restfb.com">Mark Allen</a>
954   * @since 1.6.3
955   */
956  protected static class DefaultGraphFacebookExceptionMapper implements FacebookExceptionMapper {
957    /**
958     * @see com.restfb.exception.FacebookExceptionMapper#exceptionForTypeAndMessage(java.lang.Integer,
959     *      java.lang.Integer, java.lang.String, java.lang.String)
960     */
961    public FacebookException exceptionForTypeAndMessage(Integer errorCode, Integer errorSubcode,
962        Integer httpStatusCode, String type, String message) {
963      if ("OAuthException".equals(type) || "OAuthAccessTokenException".equals(type))
964        return new FacebookOAuthException(type, message, errorCode, errorSubcode, httpStatusCode);
965
966      if ("QueryParseException".equals(type))
967        return new FacebookQueryParseException(type, message, errorCode, errorSubcode, httpStatusCode);
968
969      // Don't recognize this exception type? Just go with the standard
970      // FacebookGraphException.
971      return new FacebookGraphException(type, message, errorCode, errorSubcode, httpStatusCode);
972    }
973  }

975  /**
976   * Generate the parameter string to be included in the Facebook API request.
977   * 
978   * @param parameters
979   *          Arbitrary number of extra parameters to include in the request.
980   * @return The parameter string to include in the Facebook API request.
981   * @throws FacebookJsonMappingException
982   *           If an error occurs when building the parameter string.
983   */
984  protected String toParameterString(Parameter... parameters) {
985    if (!isBlank(accessToken))
986      parameters = parametersWithAdditionalParameter(Parameter.with(ACCESS_TOKEN_PARAM_NAME, accessToken), parameters);
987
988    if (!isBlank(accessToken) && !isBlank(appSecret))
989      parameters =
990          parametersWithAdditionalParameter(
991            Parameter.with(APP_SECRET_PROOF_PARAM_NAME, obtainAppSecretProof(accessToken, appSecret)), parameters);
992
993    parameters = parametersWithAdditionalParameter(Parameter.with(FORMAT_PARAM_NAME, "json"), parameters);
994
995    StringBuilder parameterStringBuilder = new StringBuilder();
996    boolean first = true;
997
998    for (Parameter parameter : parameters) {
999      if (first)
1000        first = false;
1001      else
1002        parameterStringBuilder.append("&");
1003
1004      parameterStringBuilder.append(urlEncode(parameter.name));
1005      parameterStringBuilder.append("=");
1006      parameterStringBuilder.append(urlEncodedValueForParameterName(parameter.name, parameter.value));
1007    }
1008
1009    return parameterStringBuilder.toString();
1010  }

1015  @Override
1016  protected String createEndpointForApiCall(String apiCall, boolean hasAttachment) {
1017    trimToEmpty(apiCall).toLowerCase();
1018    while (apiCall.startsWith("/"))
1019      apiCall = apiCall.substring(1);
1020
1021    String baseUrl = getFacebookGraphEndpointUrl();
1022
1023    if (readOnlyApiCalls.contains(apiCall))
1024      baseUrl = getFacebookReadOnlyEndpointUrl();
1025    else if (hasAttachment && apiCall.endsWith("/videos"))
1026      baseUrl = getFacebookGraphVideoEndpointUrl();
1027
1028    return format("%s/%s", baseUrl, apiCall);
1029  }
