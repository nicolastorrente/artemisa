111  /**
112   * @see com.restfb.JsonMapper#toJavaList(java.lang.String, java.lang.Class)
113   */
114  @Override
115  public <T> List<T> toJavaList(String json, Class<T> type) {
116    if (type == null)
117      throw new FacebookJsonMappingException("You must specify the Java type to map to.");
118
119    json = trimToEmpty(json);
120
121    if (isBlank(json)) {
122      if (jsonMappingErrorHandler.handleMappingError(json, type, null))
123        return null;
124      throw new FacebookJsonMappingException("JSON is an empty string - can't map it.");
125    }
126
127    if (json.startsWith("{")) {
128      // Sometimes Facebook returns the empty object {} when it really should be
129      // returning an empty list [] (example: do an FQL query for a user's
130      // affiliations - it's a list except when there are none, then it turns
131      // into an object). Check for that special case here.
132      if (isEmptyObject(json)) {
133        if (logger.isLoggable(FINER))
134          logger.finer("Encountered {} when we should've seen []. "
135              + "Mapping the {} as an empty list and moving on...");
136
137        return new ArrayList<T>();
138      }
139
140      // Special case: if the only element of this object is an array called
141      // "data", then treat it as a list. The Graph API uses this convention for
142      // connections and in a few other places, e.g. comments on the Post
143      // object.
144      // Doing this simplifies mapping, so we don't have to worry about having a
145      // little placeholder object that only has a "data" value.
146      try {
147        JsonObject jsonObject = new JsonObject(json);
148        String[] fieldNames = JsonObject.getNames(jsonObject);
149
150        if (fieldNames != null) {
151          boolean hasSingleDataProperty = fieldNames.length == 1 && "data".equals(fieldNames[0]);
152          Object jsonDataObject = jsonObject.get("data");
153
154          if (!hasSingleDataProperty && !(jsonDataObject instanceof JsonArray))
155            if (jsonMappingErrorHandler.handleMappingError(json, type, null))
156              return null;
157            else
158              throw new FacebookJsonMappingException("JSON is an object but is being mapped as a list "
159                  + "instead. Offending JSON is '" + json + "'.");
160
161          json = jsonDataObject.toString();
162        }
163      } catch (JsonException e) {
164        // Should never get here, but just in case...
165        if (jsonMappingErrorHandler.handleMappingError(json, type, e))
166          return null;
167        else
168          throw new FacebookJsonMappingException("Unable to convert Facebook response " + "JSON to a list of "
169              + type.getName() + " instances.  Offending JSON is " + json, e);
170      }
171    }
172
173    try {
174      List<T> list = new ArrayList<T>();
175
176      JsonArray jsonArray = new JsonArray(json);
177      for (int i = 0; i < jsonArray.length(); i++)
178        list.add(toJavaObject(jsonArray.get(i).toString(), type));
179
180      return unmodifiableList(list);
181    } catch (FacebookJsonMappingException e) {
182      throw e;
183    } catch (Exception e) {
184      if (jsonMappingErrorHandler.handleMappingError(json, type, e))
185        return null;
186      else
187        throw new FacebookJsonMappingException("Unable to convert Facebook response " + "JSON to a list of "
188            + type.getName() + " instances", e);
189    }
190  }

192  /**
193   * @see com.restfb.JsonMapper#toJavaObject(java.lang.String, java.lang.Class)
194   */
195  @Override
196  @SuppressWarnings("unchecked")
197  public <T> T toJavaObject(String json, Class<T> type) {
198    if (isBlank(json))
199      if (jsonMappingErrorHandler.handleMappingError(json, type, null))
200        return null;
201      else
202        throw new FacebookJsonMappingException("JSON is an empty string - can't map it.");
203
204    if (json.startsWith("["))
205      if (jsonMappingErrorHandler.handleMappingError(json, type, null))
206        return null;
207      else
208        throw new FacebookJsonMappingException("JSON is an array but is being mapped as an object "
209            + "- you should map it as a List instead. Offending JSON is '" + json + "'.");
210
211    try {
212      // Are we asked to map to JsonObject? If so, short-circuit right away.
213      if (type.equals(JsonObject.class))
214        return (T) new JsonObject(json);
215
216      List<FieldWithAnnotation<Facebook>> fieldsWithAnnotation = findFieldsWithAnnotation(type, Facebook.class);
217      Set<String> facebookFieldNamesWithMultipleMappings = facebookFieldNamesWithMultipleMappings(fieldsWithAnnotation);
218
219      // If there are no annotated fields, assume we're mapping to a built-in
220      // type. If this is actually the empty object, just return a new instance
221      // of the corresponding Java type.
222      if (fieldsWithAnnotation.size() == 0)
223        if (isEmptyObject(json)) {
224          T instance = createInstance(type);
225
226          // If there are any methods annotated with @JsonMappingCompleted,
227          // invoke them.
228          invokeJsonMappingCompletedMethods(instance);
229
230          return instance;
231        } else {
232          return toPrimitiveJavaType(json, type);
233        }
234
235      // Facebook will sometimes return the string "null".
236      // Check for that and bail early if we find it.
237      if ("null".equals(json))
238        return null;
239
240      // Facebook will sometimes return the string "false" to mean null.
241      // Check for that and bail early if we find it.
242      if ("false".equals(json)) {
243        if (logger.isLoggable(FINE))
244          logger.fine("Encountered 'false' from Facebook when trying to map to " + type.getSimpleName()
245              + " - mapping null instead.");
246        return null;
247      }
248
249      JsonObject jsonObject = new JsonObject(json);
250      T instance = createInstance(type);
251
252      if (instance instanceof JsonObject)
253        return (T) jsonObject;
254
255      // For each Facebook-annotated field on the current Java object, pull data
256      // out of the JSON object and put it in the Java object
257      for (FieldWithAnnotation<Facebook> fieldWithAnnotation : fieldsWithAnnotation) {
258        String facebookFieldName = getFacebookFieldName(fieldWithAnnotation);
259
260        if (!jsonObject.has(facebookFieldName)) {
261          if (logger.isLoggable(FINER))
262            logger.finer("No JSON value present for '" + facebookFieldName + "', skipping. JSON is '" + json + "'.");
263
264          continue;
265        }
266
267        fieldWithAnnotation.getField().setAccessible(true);
268
269        // Set the Java field's value.
270        //
271        // If we notice that this Facebook field name is mapped more than once,
272        // go into a special mode where we swallow any exceptions that occur
273        // when mapping to the Java field. This is because Facebook will
274        // sometimes return data in different formats for the same field name.
275        // See issues 56 and 90 for examples of this behavior and discussion.
276        if (facebookFieldNamesWithMultipleMappings.contains(facebookFieldName)) {
277          try {
278            fieldWithAnnotation.getField()
279              .set(instance, toJavaType(fieldWithAnnotation, jsonObject, facebookFieldName));
280          } catch (FacebookJsonMappingException e) {
281            logMultipleMappingFailedForField(facebookFieldName, fieldWithAnnotation, json);
282          } catch (JsonException e) {
283            logMultipleMappingFailedForField(facebookFieldName, fieldWithAnnotation, json);
284          }
285        } else {
286          try {
287            fieldWithAnnotation.getField()
288              .set(instance, toJavaType(fieldWithAnnotation, jsonObject, facebookFieldName));
289          } catch (Exception e) {
290            if (!jsonMappingErrorHandler.handleMappingError(json, type, e))
291              throw e;
292          }
293        }
294      }
295
296      // If there are any methods annotated with @JsonMappingCompleted,
297      // invoke them.
298      invokeJsonMappingCompletedMethods(instance);
299
300      return instance;
301    } catch (FacebookJsonMappingException e) {
302      throw e;
303    } catch (Exception e) {
304      if (jsonMappingErrorHandler.handleMappingError(json, type, e))
305        return null;
306      else
307        throw new FacebookJsonMappingException("Unable to map JSON to Java. Offending JSON is '" + json + "'.", e);
308    }
309  }

311  /**
312   * Finds and invokes methods on {@code object} that are annotated with the {@code @JsonMappingCompleted} annotation.
313   * <p>
314   * This will even work on {@code private} methods.
315   * 
316   * @param object
317   *          The object on which to invoke the method.
318   * @throws IllegalArgumentException
319   *           If unable to invoke the method.
320   * @throws IllegalAccessException
321   *           If unable to invoke the method.
322   * @throws InvocationTargetException
323   *           If unable to invoke the method.
324   */
325  protected void invokeJsonMappingCompletedMethods(Object object) throws IllegalArgumentException,
326      IllegalAccessException, InvocationTargetException {
327    for (Method method : findMethodsWithAnnotation(object.getClass(), JsonMappingCompleted.class)) {
328      method.setAccessible(true);
329
330      if (method.getParameterTypes().length == 0)
331        method.invoke(object);
332      else if (method.getParameterTypes().length == 1 && JsonMapper.class.equals(method.getParameterTypes()[0]))
333        method.invoke(object, this);
334      else
335        throw new FacebookJsonMappingException(format(
336          "Methods annotated with @%s must take 0 parameters or a single %s parameter. Your method was %s",
337          JsonMappingCompleted.class.getSimpleName(), JsonMapper.class.getSimpleName(), method));
338    }
339  }

341  /**
342   * Dumps out a log message when one of a multiple-mapped Facebook field name JSON-to-Java mapping operation fails.
343   * 
344   * @param facebookFieldName
345   *          The Facebook field name.
346   * @param fieldWithAnnotation
347   *          The Java field to map to and its annotation.
348   * @param json
349   *          The JSON that failed to map to the Java field.
350   */
351  protected void logMultipleMappingFailedForField(String facebookFieldName,
352      FieldWithAnnotation<Facebook> fieldWithAnnotation, String json) {
353    if (!logger.isLoggable(FINER))
354      return;
355
356    Field field = fieldWithAnnotation.getField();
357
358    if (logger.isLoggable(FINER))
359      logger.finer("Could not map '" + facebookFieldName + "' to " + field.getDeclaringClass().getSimpleName() + "."
360          + field.getName() + ", but continuing on because '" + facebookFieldName
361          + "' is mapped to multiple fields in " + field.getDeclaringClass().getSimpleName() + ". JSON is " + json);
362  }

364  /**
365   * For a Java field annotated with the {@code Facebook} annotation, figure out what the corresponding Facebook JSON
366   * field name to map to it is.
367   * 
368   * @param fieldWithAnnotation
369   *          A Java field annotated with the {@code Facebook} annotation.
370   * @return The Facebook JSON field name that should be mapped to this Java field.
371   */
372  protected String getFacebookFieldName(FieldWithAnnotation<Facebook> fieldWithAnnotation) {
373    String facebookFieldName = fieldWithAnnotation.getAnnotation().value();
374    Field field = fieldWithAnnotation.getField();
375
376    // If no Facebook field name was specified in the annotation, assume
377    // it's the same name as the Java field
378    if (isBlank(facebookFieldName)) {
379      if (logger.isLoggable(FINEST))
380        logger.finest("No explicit Facebook field name found for " + field
381            + ", so defaulting to the field name itself (" + field.getName() + ")");
382
383      facebookFieldName = field.getName();
384    }
385
386    return facebookFieldName;
387  }

389  /**
390   * Finds any Facebook JSON fields that are mapped to more than 1 Java field.
391   * 
392   * @param fieldsWithAnnotation
393   *          Java fields annotated with the {@code Facebook} annotation.
394   * @return Any Facebook JSON fields that are mapped to more than 1 Java field.
395   */
396  protected Set<String> facebookFieldNamesWithMultipleMappings(List<FieldWithAnnotation<Facebook>> fieldsWithAnnotation) {
397    Map<String, Integer> facebookFieldsNamesWithOccurrenceCount = new HashMap<String, Integer>();
398    Set<String> facebookFieldNamesWithMultipleMappings = new HashSet<String>();
399
400    // Get a count of Facebook field name occurrences for each
401    // @Facebook-annotated field
402    for (FieldWithAnnotation<Facebook> fieldWithAnnotation : fieldsWithAnnotation) {
403      String fieldName = getFacebookFieldName(fieldWithAnnotation);
404      int occurrenceCount =
405          facebookFieldsNamesWithOccurrenceCount.containsKey(fieldName) ? facebookFieldsNamesWithOccurrenceCount
406            .get(fieldName) : 0;
407      facebookFieldsNamesWithOccurrenceCount.put(fieldName, occurrenceCount + 1);
408    }
409
410    // Pull out only those field names with multiple mappings
411    for (Entry<String, Integer> entry : facebookFieldsNamesWithOccurrenceCount.entrySet())
412      if (entry.getValue() > 1)
413        facebookFieldNamesWithMultipleMappings.add(entry.getKey());
414
415    return unmodifiableSet(facebookFieldNamesWithMultipleMappings);
416  }

435  /**
436   * Recursively marshal the given {@code object} to JSON.
437   * <p>
438   * Used by {@link #toJson(Object)}.
439   * 
440   * @param object
441   *          The object to marshal.
442   * @param ignoreNullValuedProperties
443   *          If this is {@code true}, no Javabean properties with {@code null} values will be included in the generated
444   *          JSON.
445   * @return JSON representation of the given {@code object}.
446   * @throws FacebookJsonMappingException
447   *           If an error occurs while marshaling to JSON.
448   */
449  protected Object toJsonInternal(Object object, boolean ignoreNullValuedProperties) {
450    if (object == null)
451      return NULL;
452
453    if (object instanceof List<?>) {
454      JsonArray jsonArray = new JsonArray();
455      for (Object o : (List<?>) object)
456        jsonArray.put(toJsonInternal(o, ignoreNullValuedProperties));
457
458      return jsonArray;
459    }
460
461    if (object instanceof Map<?, ?>) {
462      JsonObject jsonObject = new JsonObject();
463      for (Entry<?, ?> entry : ((Map<?, ?>) object).entrySet()) {
464        if (!(entry.getKey() instanceof String))
465          throw new FacebookJsonMappingException("Your Map keys must be of type " + String.class
466              + " in order to be converted to JSON.  Offending map is " + object);
467
468        try {
469          jsonObject.put((String) entry.getKey(), toJsonInternal(entry.getValue(), ignoreNullValuedProperties));
470        } catch (JsonException e) {
471          throw new FacebookJsonMappingException("Unable to process value '" + entry.getValue() + "' for key '"
472              + entry.getKey() + "' in Map " + object, e);
473        }
474      }
475
476      return jsonObject;
477    }
478
479    if (isPrimitive(object))
480      return object;
481
482    if (object instanceof BigInteger)
483      return ((BigInteger) object).longValue();
484
485    if (object instanceof BigDecimal)
486      return ((BigDecimal) object).doubleValue();
487
488    // We've passed the special-case bits, so let's try to marshal this as a
489    // plain old Javabean...
490
491    List<FieldWithAnnotation<Facebook>> fieldsWithAnnotation =
492        findFieldsWithAnnotation(object.getClass(), Facebook.class);
493
494    JsonObject jsonObject = new JsonObject();
495
496    // No longer throw an exception in this case. If there are multiple fields
497    // with the same @Facebook value, it's luck of the draw which is picked for
498    // JSON marshaling.
499    // TODO: A better implementation would query each duplicate-mapped field. If
500    // it has is a non-null value and the other duplicate values are null, use
501    // the non-null field.
502    Set<String> facebookFieldNamesWithMultipleMappings = facebookFieldNamesWithMultipleMappings(fieldsWithAnnotation);
503    if (facebookFieldNamesWithMultipleMappings.size() > 0 && logger.isLoggable(FINE))
504      logger.fine(format("Unable to convert to JSON because multiple @" + Facebook.class.getSimpleName()
505          + " annotations for the same name are present: " + facebookFieldNamesWithMultipleMappings));
506
507    for (FieldWithAnnotation<Facebook> fieldWithAnnotation : fieldsWithAnnotation) {
508      String facebookFieldName = getFacebookFieldName(fieldWithAnnotation);
509      fieldWithAnnotation.getField().setAccessible(true);
510
511      try {
512        Object fieldValue = fieldWithAnnotation.getField().get(object);
513
514        if (!(ignoreNullValuedProperties && fieldValue == null))
515          jsonObject.put(facebookFieldName, toJsonInternal(fieldValue, ignoreNullValuedProperties));
516      } catch (Exception e) {
517        throw new FacebookJsonMappingException("Unable to process field '" + facebookFieldName + "' for "
518            + object.getClass(), e);
519      }
520    }
521
522    return jsonObject;
523  }

525  /**
526   * Given a {@code json} value of something like {@code MyValue} or {@code 123} , return a representation of that value
527   * of type {@code type}.
528   * <p>
529   * This is to support non-legal JSON served up by Facebook for API calls like {@code Friends.get} (example result:
530   * {@code [222333,1240079]}).
531   * 
532   * @param <T>
533   *          The Java type to map to.
534   * @param json
535   *          The non-legal JSON to map to the Java type.
536   * @param type
537   *          Type token.
538   * @return Java representation of {@code json}.
539   * @throws FacebookJsonMappingException
540   *           If an error occurs while mapping JSON to Java.
541   */
542  @SuppressWarnings("unchecked")
543  protected <T> T toPrimitiveJavaType(String json, Class<T> type) {
544
545    if (String.class.equals(type)) {
546      // If the string starts and ends with quotes, remove them, since Facebook
547      // can serve up strings surrounded by quotes.
548      if (json.length() > 1 && json.startsWith("\"") && json.endsWith("\"")) {
549        json = json.replaceFirst("\"", "");
550        json = json.substring(0, json.length() - 1);
551      }
552
553      return (T) json;
554    }
555
556    if (Integer.class.equals(type) || Integer.TYPE.equals(type))
557      return (T) new Integer(json);
558    if (Boolean.class.equals(type) || Boolean.TYPE.equals(type))
559      return (T) new Boolean(json);
560    if (Long.class.equals(type) || Long.TYPE.equals(type))
561      return (T) new Long(json);
562    if (Double.class.equals(type) || Double.TYPE.equals(type))
563      return (T) new Double(json);
564    if (Float.class.equals(type) || Float.TYPE.equals(type))
565      return (T) new Float(json);
566    if (BigInteger.class.equals(type))
567      return (T) new BigInteger(json);
568    if (BigDecimal.class.equals(type))
569      return (T) new BigDecimal(json);
570
571    if (jsonMappingErrorHandler.handleMappingError(json, type, null))
572      return null;
573
574    throw new FacebookJsonMappingException("Don't know how to map JSON to " + type
575        + ". Are you sure you're mapping to the right class? " + "Offending JSON is '" + json + "'.");
576  }

578  /**
579   * Extracts JSON data for a field according to its {@code Facebook} annotation and returns it converted to the proper
580   * Java type.
581   * 
582   * @param fieldWithAnnotation
583   *          The field/annotation pair which specifies what Java type to convert to.
584   * @param jsonObject
585   *          "Raw" JSON object to pull data from.
586   * @param facebookFieldName
587   *          Specifies what JSON field to pull "raw" data from.
588   * @return A
589   * @throws JsonException
590   *           If an error occurs while mapping JSON to Java.
591   * @throws FacebookJsonMappingException
592   *           If an error occurs while mapping JSON to Java.
593   */
594  protected Object toJavaType(FieldWithAnnotation<Facebook> fieldWithAnnotation, JsonObject jsonObject,
595      String facebookFieldName) {
596    Class<?> type = fieldWithAnnotation.getField().getType();
597    Object rawValue = jsonObject.get(facebookFieldName);
598
599    // Short-circuit right off the bat if we've got a null value.
600    if (NULL.equals(rawValue))
601      return null;
602
603    if (String.class.equals(type)) {
604      // Special handling here for better error checking.
605      // Since JsonObject.getString() will return literal JSON text even if it's
606      // _not_ a JSON string, we check the marshaled type and bail if needed.
607      // For example, calling JsonObject.getString("results") on the below
608      // JSON...
609      // {"results":[{"name":"Mark Allen"}]}
610      // ... would return the string "[{"name":"Mark Allen"}]" instead of
611      // throwing an error. So we throw the error ourselves.
612
613      // Per Antonello Naccarato, sometimes FB will return an empty JSON array
614      // instead of an empty string. Look for that here.
615      if (rawValue instanceof JsonArray)
616        if (((JsonArray) rawValue).length() == 0) {
617          if (logger.isLoggable(FINER))
618            logger.finer("Coercing an empty JSON array " + "to an empty string for " + fieldWithAnnotation);
619
620          return "";
621        }
622
623      // If the user wants a string, _always_ give her a string.
624      // This is useful if, for example, you've got a @Facebook-annotated string
625      // field that you'd like to have a numeric type shoved into.
626      // User beware: this will turn *anything* into a string, which might lead
627      // to results you don't expect.
628      return rawValue.toString();
629    }
630
631    if (Integer.class.equals(type) || Integer.TYPE.equals(type))
632      return new Integer(jsonObject.getInt(facebookFieldName));
633    if (Boolean.class.equals(type) || Boolean.TYPE.equals(type))
634      return new Boolean(jsonObject.getBoolean(facebookFieldName));
635    if (Long.class.equals(type) || Long.TYPE.equals(type))
636      return new Long(jsonObject.getLong(facebookFieldName));
637    if (Double.class.equals(type) || Double.TYPE.equals(type))
638      return new Double(jsonObject.getDouble(facebookFieldName));
639    if (Float.class.equals(type) || Float.TYPE.equals(type))
640      return new BigDecimal(jsonObject.getString(facebookFieldName)).floatValue();
641    if (BigInteger.class.equals(type))
642      return new BigInteger(jsonObject.getString(facebookFieldName));
643    if (BigDecimal.class.equals(type))
644      return new BigDecimal(jsonObject.getString(facebookFieldName));
645    if (List.class.equals(type))
646      return toJavaList(rawValue.toString(), getFirstParameterizedTypeArgument(fieldWithAnnotation.getField()));
647
648    String rawValueAsString = rawValue.toString();
649
650    // Hack for issue 76 where FB will sometimes return a Post's Comments as
651    // "[]" instead of an object type (wtf)
652    if (Comments.class.isAssignableFrom(type) && rawValue instanceof JsonArray) {
653      if (logger.isLoggable(FINE))
654        logger.fine("Encountered comment array '" + rawValueAsString + "' but expected a "
655            + Comments.class.getSimpleName() + " object instead.  Working around that " + "by coercing into an empty "
656            + Comments.class.getSimpleName() + " instance...");
657
658      JsonObject workaroundJsonObject = new JsonObject();
659      workaroundJsonObject.put("total_count", 0);
660      workaroundJsonObject.put("data", new JsonArray());
661      rawValueAsString = workaroundJsonObject.toString();
662    }
663
664    // Some other type - recurse into it
665    return toJavaObject(rawValueAsString, type);
666  }

668  /**
669   * Creates a new instance of the given {@code type}.
670   * <p>
671   * 
672   * 
673   * @param <T>
674   *          Java type to map to.
675   * @param type
676   *          Type token.
677   * @return A new instance of {@code type}.
678   * @throws FacebookJsonMappingException
679   *           If an error occurs when creating a new instance ({@code type} is inaccessible, doesn't have a no-arg
680   *           constructor, etc.)
681   */
682  protected <T> T createInstance(Class<T> type) {
683    String errorMessage =
684        "Unable to create an instance of " + type
685            + ". Please make sure that if it's a nested class, is marked 'static'. "
686            + "It should have a no-argument constructor.";
687
688    try {
689      Constructor<T> defaultConstructor = type.getDeclaredConstructor();
690
691      if (defaultConstructor == null)
692        throw new FacebookJsonMappingException("Unable to find a default constructor for " + type);
693
694      // Allows protected, private, and package-private constructors to be
695      // invoked
696      defaultConstructor.setAccessible(true);
697      return defaultConstructor.newInstance();
698    } catch (Exception e) {
699      throw new FacebookJsonMappingException(errorMessage, e);
700    }
701  }

703  /**
704   * Is the given JSON equivalent to the empty object (<code>{}</code>)?
705   * 
706   * @param json
707   *          The JSON to check.
708   * @return {@code true} if the JSON is equivalent to the empty object, {@code false} otherwise.
709   */
710  protected boolean isEmptyObject(String json) {
711    return "{}".equals(json);
712  }

716   * 
717   * @author <a href="http://restfb.com">Mark Allen</a>
718   * @since 1.6.2
719   */
720  public static interface JsonMappingErrorHandler {
721    /**
722     * This method will be called by {@code DefaultJsonMapper} if it encounters an error while attempting to map JSON to
723     * a Java object.
724     * <p>
725     * You may perform any behavior you'd like here in response to an error, e.g. logging it.
726     * <p>
727     * If the mapper should continue processing, return {@code true} and {@code null} will be mapped to the target type.
728     * If you would like the mapper to stop processing and throw
729     * {@link com.restfb.exception.FacebookJsonMappingException}, return {@code false}.
730     * 
731     * @param unmappableJson

733     * @param targetType

735     * @param e

737     *          exception.
738     * @return {@code true} to continue processing, {@code false} to throw a
739     *         {@link com.restfb.exception.FacebookJsonMappingException}.
740     */
741    boolean handleMappingError(String unmappableJson, Class<?> targetType, Exception e);
