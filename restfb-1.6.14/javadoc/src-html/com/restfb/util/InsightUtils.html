062  public enum Period {
063    DAY(60 * 60 * 24), WEEK(60 * 60 * 24 * 7), DAYS_28(60 * 60 * 24 * 28), MONTH(2592000), LIFETIME(0);

071    public int getPeriodLength() {
072      return periodLength;
073    }

079  /**
080   * Queries Facebook via FQL for several Insights at different date points.
081   * <p>
082   * The output groups the result by metric and then by date, matching the input arguments. Map entries may be
083   * {@code null} if Facebook does not return corresponding data, e.g. when you query a metric which is not available at
084   * the chosen period. The inner {@code Map}'s {@code Object} value may be a strongly typed value for some metrics, but
085   * in other cases Facebook returns a {@code JsonObject}.
086   * <p>
087   * Sample output, assuming 2 metrics were queried for 5 dates:
088   * 
089   * <pre>
090   * {page_active_users = {
091   * &nbsp;&nbsp;&nbsp; 2011-jan-01 = 7,
092   * &nbsp;&nbsp;&nbsp; 2011-jan-02 = 26,
093   * &nbsp;&nbsp;&nbsp; 2011-jan-03 = 15,
094   * &nbsp;&nbsp;&nbsp; 2011-jan-04 = 10,
095   * &nbsp;&nbsp;&nbsp; 2011-jan-05= 687},
096   * page_tab_views_login_top_unique = {
097   * &nbsp;&nbsp;&nbsp; 2011-jan-01 = {"photos":2,"app_4949752878":3,"wall":30},
098   * &nbsp;&nbsp;&nbsp; 2011-jan-02 = {"app_4949752878":1,"photos":1,"app_2373072738":2,"wall":23},
099   * &nbsp;&nbsp;&nbsp; 2011-jan-03 = {"app_4949752878":1,"wall":12},
100   * &nbsp;&nbsp;&nbsp; 2011-jan-04 = {"photos":1,"wall":11},
101   * &nbsp;&nbsp;&nbsp; 2011-jan-05 = {"app_494975287":2,"app_237307273":2,"photos":6,"wall":35,"info":6}}
102   * </pre>
103   * 
104   * @param facebookClient
105   *          The {@code FacebookClient} used to communicate with the Insights API.
106   * @param pageObjectId
107   *          The required object_id to query.
108   * @param metrics
109   *          A not null/empty set of metrics that will be queried for the given {@code period}.
110   * @param period
111   *          The required time period over which to query.
112   * @param periodEndDates
113   *          A non-null, non-empty {@code Set} in which each date should be normalized to be midnight in the PST
114   *          timezone; see {@link #convertToMidnightInPacificTimeZone(Set)}.
115   * @return A {@code Map} of {@code Maps}: the outer keys will be all the metrics requested that were available at the
116   *         period/times requested. The inner keys will be the {@code Set} of periodEndDates.
117   * @see #convertToMidnightInPacificTimeZone(Set)
118   * @see #executeInsightQueriesByDate(FacebookClient, String, Set, Period, Set)
119   */
120  public static SortedMap<String, SortedMap<Date, Object>> executeInsightQueriesByMetricByDate(
121      FacebookClient facebookClient, String pageObjectId, Set<String> metrics, Period period, Set<Date> periodEndDates) {
122    SortedMap<String, SortedMap<Date, Object>> result = new TreeMap<String, SortedMap<Date, Object>>();
123    SortedMap<Date, JsonArray> raw =
124        executeInsightQueriesByDate(facebookClient, pageObjectId, metrics, period, periodEndDates);
125
126    if (!raw.isEmpty()) {
127      for (Date date : raw.keySet()) {
128        JsonArray resultByMetric = raw.get(date);
129
130        // [{"metric":"page_active_users","value":582},
131        // {"metric":"page_tab_views_login_top_unique","value":{"wall":12,"app_4949752878":1}}]
132        for (int resultIndex = 0; resultIndex < resultByMetric.length(); resultIndex++) {
133          JsonObject metricResult = resultByMetric.getJsonObject(resultIndex);
134
135          try {
136            String metricName = metricResult.getString("metric");
137            Object metricValue = metricResult.get("value");
138
139            // store into output collection
140            SortedMap<Date, Object> resultByDate = result.get(metricName);
141            if (resultByDate == null) {
142              resultByDate = new TreeMap<Date, Object>();
143              result.put(metricName, resultByDate);
144            }
145
146            if (resultByDate.put(date, metricValue) != null)
147              throw new IllegalStateException(format(
148                "MultiQuery response has two results for metricName: %s and date: %s", metricName, date));
149          } catch (JsonException e) {
150            throw new FacebookJsonMappingException(format("Could not decode result for %s: %s", metricResult,
151              e.getMessage()), e);
152          }
153        }
154      }
155    }
156
157    return result;
158  }

160  /**
161   * Queries Facebook via FQL for several Insights at different date points and returns "raw" results.
162   * <p>
163   * A variation on {@link #executeInsightQueriesByMetricByDate(FacebookClient, String, Set, Period, Set)} , this method
164   * returns the raw output from the Facebook, keying the output by date alone. The {@code JsonArray} value will contain
165   * the metrics that were requested and available at the date.
166   * <p>
167   * Sample output, assuming 2 metrics were queried for 5 dates:
168   * 
169   * <pre>
170   * {2011-jan-01 = [
171   * &nbsp;&nbsp;&nbsp; {"metric":"page_active_users","value":7},
172   * &nbsp;&nbsp;&nbsp; {"metric":"page_tab_views_login_top_unique","value":{"photos":2,"app_4949752878":3,"wall":30}}], 
173   * 2011-jan-02 = [
174   * &nbsp;&nbsp;&nbsp; {"metric":"page_active_users","value":26},
175   * &nbsp;&nbsp;&nbsp; {"metric":"page_tab_views_login_top_unique","value":{"app_4949752878":1,"photos":1,"app_2373072738":2,"wall":23}}], 
176   * 2011-jan-03 = [
177   * &nbsp;&nbsp;&nbsp; {"metric":"page_active_users","value":15},
178   * &nbsp;&nbsp;&nbsp; {"metric":"page_tab_views_login_top_unique","value":{"app_4949752878":1,"wall":12}}], 
179   * 2011-jan-04 = [
180   * &nbsp;&nbsp;&nbsp; {"metric":"page_active_users","value":10},
181   * &nbsp;&nbsp;&nbsp; {"metric":"page_tab_views_login_top_unique","value":{"photos":1,"wall":11}}]
182   * 2011-jan-05 = [
183   * &nbsp;&nbsp;&nbsp; {"metric":"page_active_users","value":687},
184   * &nbsp;&nbsp;&nbsp; {"metric":"page_tab_views_login_top_unique","value":{"app_494975287":2,"app_237307273":2,"photos":6,"wall":35,"info":6}}]
185   * }
186   * </pre>
187   * 
188   * @param facebookClient
189   *          The {@code FacebookClient} used to communicate with the Insights API.
190   * @param pageObjectId
191   *          The required object_id to query.
192   * @param metrics
193   *          A not null/empty set of metrics that will be queried for the given {@code period}.
194   * @param period
195   *          The required time period over which to query.
196   * @param periodEndDates
197   *          A non-null, non-empty {@code Set} in which each date should be normalized to be midnight in the PST
198   *          timezone; see {@link #convertToMidnightInPacificTimeZone(Set)}.
199   * @return Insights query results, grouped by date.
200   * @see #convertToMidnightInPacificTimeZone(Set)
201   * @see #executeInsightQueriesByMetricByDate(FacebookClient, String, Set, Period, Set)
202   */
203  public static SortedMap<Date, JsonArray> executeInsightQueriesByDate(FacebookClient facebookClient,
204      String pageObjectId, Set<String> metrics, Period period, Set<Date> periodEndDates) {
205    if (facebookClient == null)
206      throw new IllegalArgumentException("The 'facebookClient' parameter is required.");
207
208    if (isBlank(pageObjectId))
209      throw new IllegalArgumentException(
210        "The 'pageObjectId' parameter should be a non-empty string, probably a positive number.");
211
212    if (isEmpty(metrics))
213      throw new IllegalArgumentException("The 'metrics' set should be non-empty.");
214
215    if (period == null)
216      throw new IllegalArgumentException("The 'period' parameter is required.");
217
218    if (isEmpty(periodEndDates))
219      throw new IllegalArgumentException("The 'periodEndDates' set should be non-empty");
220
221    // put dates into an array where we can easily access the index of each
222    // date, as these ordinal positions will be used as query identifiers
223    // in the MultiQuery
224    List<Date> datesByQueryIndex = new ArrayList<Date>(periodEndDates);
225    // sort the list in Date order so the implementation is tolerant of a
226    // chaotic periodEndDates iteration order
227    Collections.sort(datesByQueryIndex);
228
229    String baseQuery = createBaseQuery(period, pageObjectId, metrics);
230
231    Map<String, String> fqlByQueryIndex = buildQueries(baseQuery, datesByQueryIndex);
232
233    SortedMap<Date, JsonArray> result = new TreeMap<Date, JsonArray>();
234
235    if (!fqlByQueryIndex.isEmpty()) {
236      // request the client sends all the queries in fqlByQueryIndex to Facebook
237      // in one go and have the raw JsonObject be returned
238      JsonObject response = facebookClient.executeFqlMultiquery(fqlByQueryIndex, JsonObject.class);
239
240      // transform the response into a Map
241      for (Iterator<?> it = response.keys(); it.hasNext();) {
242        String key = (String) it.next();
243        JsonArray innerResult = (JsonArray) response.get(key);
244
245        try {
246          // resolve the map key back into a date
247          int queryIndex = Integer.parseInt(key);
248          Date d = datesByQueryIndex.get(queryIndex);
249          if (d == null)
250            throw new IllegalStateException("MultiQuery response had an unexpected key value: " + key);
251
252          result.put(d, innerResult);
253        } catch (NumberFormatException nfe) {
254          throw new IllegalStateException("MultiQuery response had an unexpected key value: " + key, nfe);
255        }
256      }
257    }
258    return result;
259  }
260
261  static Map<String, String> buildQueries(String baseQuery, List<Date> datesByQueryIndex) {
262    Map<String, String> fqlByQueryIndex = new LinkedHashMap<String, String>();
263    for (int queryIndex = 0; queryIndex < datesByQueryIndex.size(); queryIndex++) {
264      Date d = datesByQueryIndex.get(queryIndex);
265      String query = baseQuery + convertToUnixTimeOneDayLater(d);
266      fqlByQueryIndex.put(String.valueOf(queryIndex), query);
267    }
268    return fqlByQueryIndex;
269  }
270
271  static String createBaseQuery(Period period, String pageObjectId, Set<String> metrics) {
272    StringBuilder q = new StringBuilder();
273    q.append("SELECT metric, value ");
274    q.append("FROM insights ");
275    q.append("WHERE object_id='");
276    q.append(pageObjectId);
277    q.append('\'');
278
279    String metricInList = buildMetricInList(metrics);
280    if (!isBlank(metricInList)) {
281      q.append(" AND metric IN (");
282      q.append(metricInList);
283      q.append(")");
284    }
285
286    q.append(" AND period=");
287    q.append(period.getPeriodLength());
288    q.append(" AND end_time=");
289    return q.toString();
290  }
291
292  static String buildMetricInList(Set<String> metrics) {
293    StringBuilder in = new StringBuilder();
294    if (!isEmpty(metrics)) {
295      int metricCount = 0;
296      for (String metric : metrics) {
297        if (!isBlank(metric)) {
298          if (metricCount > 0)
299            in.append(',');
300
301          in.append('\'');
302          in.append(metric.trim());
303          in.append('\'');
304          metricCount++;
305        }
306      }
307    }
308    return in.toString();
309  }

311  /**
312   * Slides this date back to midnight in the PST timezone fit for the Facebook Query Language.
313   * <p>
314   * More details are available at <a target="_blank" href="http://developers.facebook.com/docs/reference/fql/insights"
315   * >http://developers.facebook.com/docs/reference/fql/insights</a>.
316   * 
317   * @param date
318   *          The date to slide back.
319   * @return A midnight-PST version of the provided {@code date}.
320   * @see #convertToMidnightInPacificTimeZone(Set)
321   */
322  public static Date convertToMidnightInPacificTimeZone(Date date) {
323    if (date == null)
324      throw new IllegalArgumentException("The 'date' parameter is required.");
325
326    Set<Date> convertedDates = convertToMidnightInPacificTimeZone(singleton(date));
327
328    if (convertedDates.size() != 1)
329      throw new IllegalStateException("Internal error, expected 1 date.");
330
331    return convertedDates.iterator().next();
332  }

334  /**
335   * Slides these dates back to midnight in the PST timezone fit for the Facebook Query Language.
336   * <p>
337   * More details are available at <a target="_blank" href="http://developers.facebook.com/docs/reference/fql/insights"
338   * >http://developers.facebook.com/docs/reference/fql/insights</a>.
339   * 
340   * @param dates
341   *          The dates to slide back.
342   * @return Midnight-PST versions of the provided {@code dates}.
343   * @see #convertToMidnightInPacificTimeZone(Date)
344   */
345  public static SortedSet<Date> convertToMidnightInPacificTimeZone(Set<Date> dates) {
346    Calendar calendar = Calendar.getInstance(PST_TIMEZONE);
347    SortedSet<Date> convertedDates = new TreeSet<Date>();
348    for (Date d : dates) {
349      calendar.setTime(d);
350      calendar.set(Calendar.HOUR_OF_DAY, 0);
351      calendar.set(Calendar.MINUTE, 0);
352      calendar.set(Calendar.SECOND, 0);
353      calendar.set(Calendar.MILLISECOND, 0);
354      convertedDates.add(calendar.getTime());
355    }
356    return convertedDates;
357  }

359  /**
360   * Converts into a "unix time", which means convert into the number of seconds (NOT milliseconds) from the Epoch fit
361   * for the Facebook Query Language. Notice that if you want data for September 15th then you need to present to
362   * Facebook the NEXT DAY, ie. the upper exclusive limit of your date range. So beyond all the sliding to midnight code
363   * you see in {@link #convertToMidnightInPacificTimeZone(Date)}, we need to go further and slide this input date
364   * forward one day.
365   * 
366   * In retrospect, this should have been implemented via the Facebook end_time_date() function.
367   * 
368   * @param date
369   *          The date to convert.
370   * @return Unix time representation of the given {@code date}.
371   */
372  static long convertToUnixTimeOneDayLater(Date date) {
373    long time = date.getTime() / 1000L;
374    // note we cannot use a Daylight sensitive Calendar here since that would
375    // adjust the time incorrectly over the DST junction
376    time += SECONDS_IN_DAY;
377    return time;
378  }

380  /**
381   * Slides this time back to midnight in the PST timezone and converts into a "unix time", which means convert into the
382   * number of seconds (NOT milliseconds) from the Epoch fit for the Facebook Query Language. Notice that if you want
383   * data for September 15th then you need to present to Facebook the NEXT DAY, ie. the upper exclusive limit of your
384   * date range. So beyond all the sliding to midnight issue, we need to go further and slide this input date forward
385   * one day.
386   * 
387   * @param date
388   *          The date to convert.
389   * @return Midnight-PST Unix time representation of the given {@code date}.
390   * @see #convertToMidnightInPacificTimeZone(Date)
391   */
392  static long convertToUnixTimeAtPacificTimeZoneMidnightOneDayLater(Date date) {
393    return convertToUnixTimeOneDayLater(convertToMidnightInPacificTimeZone(date));
394  }
