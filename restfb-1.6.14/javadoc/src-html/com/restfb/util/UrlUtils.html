023package com.restfb.util;
024
025import static java.lang.String.format;
026import static java.net.URLDecoder.decode;
027import static java.net.URLEncoder.encode;
028import static java.util.Collections.emptyMap;
029
030import java.io.UnsupportedEncodingException;
031import java.util.ArrayList;
032import java.util.HashMap;
033import java.util.List;
034import java.util.Map;
035
036/**
037 * @author <a href="http://restfb.com">Mark Allen</a>
038 * @since 1.6.10
039 */
040public final class UrlUtils {
041  /**
042   * Default charset to use for encoding/decoding strings.
043   */
044  public static final String ENCODING_CHARSET = "UTF-8";
045
046  /**
047   * Prevents instantiation.
048   */
049  private UrlUtils() {}
050
051  /**
052   * URL-encodes a string.
053   * <p>
054   * Assumes {@code string} is in {@value #ENCODING_CHARSET} format.
055   * 
056   * @param string
057   *          The string to URL-encode.
058   * @return The URL-encoded version of the input string, or {@code null} if {@code string} is {@code null}.
059   * @throws IllegalStateException
060   *           If unable to URL-encode because the JVM doesn't support {@value #ENCODING_CHARSET}.
061   */
062  public static String urlEncode(String string) {
063    if (string == null)
064      return null;
065    try {
066      return encode(string, ENCODING_CHARSET);
067    } catch (UnsupportedEncodingException e) {
068      throw new IllegalStateException("Platform doesn't support " + ENCODING_CHARSET, e);
069    }
070  }
071
072  /**
073   * URL-decodes a string.
074   * <p>
075   * Assumes {@code string} is in {@value #ENCODING_CHARSET} format.
076   * 
077   * @param string
078   *          The string to URL-decode.
079   * @return The URL-decoded version of the input string, or {@code null} if {@code string} is {@code null}.
080   * @throws IllegalStateException
081   *           If unable to URL-decode because the JVM doesn't support {@value #ENCODING_CHARSET}.
082   * @since 1.6.5
083   */
084  public static String urlDecode(String string) {
085    if (string == null)
086      return null;
087    try {
088      return decode(string, ENCODING_CHARSET);
089    } catch (UnsupportedEncodingException e) {
090      throw new IllegalStateException("Platform doesn't support " + ENCODING_CHARSET, e);
091    }
092  }
093
094  /**
095   * For the given {@code queryString}, extract a mapping of query string parameter names to values.
096   * <p>
097   * Example of a {@code queryString} is {@code accessToken=123&expires=345}.
098   * 
099   * @param queryString
100   *          The URL query string from which parameters are extracted.
101   * @return A mapping of query string parameter names to values. If {@code queryString} is {@code null}, an empty
102   *         {@code Map} is returned.
103   * @throws IllegalStateException
104   *           If unable to URL-decode because the JVM doesn't support {@value #ENCODING_CHARSET}.
105   */
106  public static Map<String, List<String>> extractParametersFromQueryString(String queryString) {
107    if (queryString == null)
108      return emptyMap();
109
110    // If there is no ? character at the front of the string, append it.
111    return extractParametersFromUrl(format("restfb://url%s", queryString.startsWith("?") ? queryString : "?"
112        + queryString));
113  }
114
115  /**
116   * For the given {@code url}, extract a mapping of query string parameter names to values.
117   * <p>
118   * Adapted from an implementation by BalusC and dfrankow, available at
119   * http://stackoverflow.com/questions/1667278/parsing-query-strings-in-java.
120   * 
121   * @param url
122   *          The URL from which parameters are extracted.
123   * @return A mapping of query string parameter names to values. If {@code url} is {@code null}, an empty {@code Map}
124   *         is returned.
125   * @throws IllegalStateException
126   *           If unable to URL-decode because the JVM doesn't support {@value #ENCODING_CHARSET}.
127   */
128  public static Map<String, List<String>> extractParametersFromUrl(String url) {
129    if (url == null)
130      return emptyMap();
131
132    Map<String, List<String>> parameters = new HashMap<String, List<String>>();
133
134    String[] urlParts = url.split("\\?");
135
136    if (urlParts.length > 1) {
137      String query = urlParts[1];
138
139      for (String param : query.split("&")) {
140        String pair[] = param.split("=");
141        String key = urlDecode(pair[0]);
142        String value = "";
143
144        if (pair.length > 1)
145          value = urlDecode(pair[1]);
146
147        List<String> values = parameters.get(key);
148
149        if (values == null) {
150          values = new ArrayList<String>();
151          parameters.put(key, values);
152        }
153
154        values.add(value);
155      }
156    }
157
158    return parameters;
159  }
160}
